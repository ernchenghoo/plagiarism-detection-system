import java.util.*;
import java.io.*;
import java.text.*;

public class myshell_00000014 {
    public static void main(String[] args) throws IOException {
        String commandLine;
        ArrayList<String> commandHist = new ArrayList<String>();
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("My Shell Version 0.01");

        while (true) {
            System.out.print("00000014> ");
            commandLine = console.readLine();

            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after splitting them.
            ArrayList<String> commandParams = new ArrayList<String>();
            String[] commandSplit = commandLine.split(" ");

            if (commandSplit[0].equals("")) {
                continue;
            }

            if (!commandLine.isEmpty()) {
                commandHist.add(commandLine);
            }

            //Check to see if there is any parameter
            if (commandSplit.length > 1) {
                //Add the parameters (if any) into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            /* Takes the command from the commandSplit array to be used
            in the switch statement */
            String commandString = commandSplit[0];
            switch (commandString) {

                case "about":
                    System.out.println("Author: Tang Kar Chuin");
                    System.out.println("Version: 0.01");
                    break;

                case "time":
                    Date time = new Date();
                    //Format the output in hour:minute:second AM/PM format
                    SimpleDateFormat sdfTime = new SimpleDateFormat("hh:mm:ss a");
                    System.out.println(sdfTime.format(time));
                    break;

                case "date":
                    Date date = new Date();
                    //Format the output in day/month/year format
                    SimpleDateFormat sdfDate = new SimpleDateFormat("E dd/MM/yyyy");
                    System.out.println(sdfDate.format(date));
                    break;

                case "notepad":
                    ProcessBuilder pb = new ProcessBuilder("C:\\Windows\\system32\\notepad.exe");
                    pb.start();
                    break;

                case "exit":
                    System.exit(0);
                    break;

                case "ls":
                    File f = new File(".");
                    File[] filesList = f.listFiles();

                    if (commandParams.size() < 1) {
                        for (File i : filesList) {
                            if (i.isDirectory()) {
                                System.out.println(i.getName());
                            }
                            if (i.isFile()) {
                                System.out.println(i.getName());
                            }
                        }
                    } else 
if (commandParams.get(0).equals("-f")) {
                        for (File i : filesList) {
                            if (i.isFile()) System.out.println(i.getName());
                        }
                    } else 
if (commandParams.get(0).equals("-d")) {
                        for (File i : filesList) {
                            if (i.isDirectory()) System.out.println(i.getName());
                        }
                    }
                    break;

                case "hist":
                    if (commandParams.size() < 1) {
                        System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    } else {

                        if(commandParams.size() == 3){

                            if (commandParams.get(0).equals("-c")) {
                                commandHist.clear();
                                System.out.println("Deleted: All command history.");

                            } else {

                                //Parses the number after -l argument into an integer
                                String histCArgNum = commandParams.get(1);
                                int histCInput = Integer.valueOf(histCArgNum);

                                if (histCInput < 1 || histCInput > commandHist.size()) {
                                    System.out.println("Invalid number");
                                } else {
                                    for (int i = 0; i <= histCInput - 1; i++) {
                                        System.out.println(commandHist.get(i));
                                    }
                                }

                                commandHist.clear();
                                System.out.println("Deleted: All command history.");
                            }

                        } else {
                            //Do the shit at the bottom


                            //Process based on the number of parameters attached to the command.
                            if (commandParams.get(0).equals("-h")) {
                                System.out.println("hist -l [int] : Retrieves the last [int] commands entered by the user.");
                                System.out.println("hist -c : Clear the command history.");
                            }

                            if (commandParams.get(0).equals("-c")) {
                                commandHist.clear();
                                
System.out.println("Deleted: All command history.");
                            }

                            if (commandParams.get(0).equals("-l")) {

                                //Parses the number after -l argument into an integer
                                String histCArgNum = commandParams.get(1);
                                int histCInput = Integer.valueOf(histCArgNum);

                                if (histCInput < 1 || histCInput > commandHist.size()) {
                                    System.out.println("Invalid number");
                                } else {
                                    
for (int i = 0; i <= histCInput - 1; i++) {
                                        System.out.println(commandHist.get(i));
                                    }
                                }
                            }
                        } }
                    break;
                default:
                    System.out.println("Invalid command");
            }

        }
    }
}
