import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000106 {
    
    public static void main(String[] args) throws IOException {

        String CommLine;

        BufferedReader ConVers = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("My Shell Version 0.01");
        
        ArrayList CommHist = new ArrayList();
      
        
        
        while (true) {
            int x = 0;
            System.out.print("00000106> ");

            CommLine = ConVers.readLine();

            ArrayList CommParam = new ArrayList();
            String[] CommSplit = CommLine.split(" ");

            if (CommSplit.length > 1) {
                
                for (int i = 1; i < CommSplit.length; i++) {
                    CommParam.add(CommSplit[i]);
                }
            }
              
            if (CommSplit[0].equals("")) {
                continue;
            }
            
            if (CommSplit[0].equals("about")){
                System.out.println("Author: Jeremy Tay Chien Lee");
                System.out.println("Version: 0.01");
                x++;
            }
            
            if (CommSplit[0].equals("time")) {
                Date Time = new Date();
                SimpleDateFormat time = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(time.format(Time));
                x++;
            }
            
            if (CommSplit[0].equals("date")) {
                Date Date = new Date();
                SimpleDateFormat date = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(date.format(Date));
                x++;
            }
            
            if (CommSplit[0].equals("Exit")) {
                System.exit(0);
            }
            
            if (CommSplit[0].equals("ls")) {
                
if (CommParam.size() < 1){
                    x++;
                    System.out.println("List: ");
                     File files = new File(".");
                        File[] fl = files.listFiles();
                        for (File file : fl) {
                            if (file.isFile() || file.isDirectory()) {
                                System.out.println(file.getName());
                            }
                            
                        }
                }
                else{
                    if (CommParam.get(0).equals("-f")){
                        x++;
                        System.out.println("Files Lists: ");
                        File files = new File(".");
                        File[] fl = files.listFiles();
                        for (File file : fl) {
                            if (file.isFile()) {
                                System.out.println(file.getName());
                            }
                        }
                    }
                    if (CommParam.get(0).equals("-d")){
                        x++;
                        System.out.println("Directories Lists: ");
                        File files = new File(".");
                        File[] fl = files.listFiles();
                        for (File file : fl) {
                            if (file.isDirectory()) {
                                System.out.println(file.getName());
                            }
                        }
                    }
                }
            }
            
            if (CommSplit[0].equals("notepad")){
                ProcessBuilder Notepad = new ProcessBuilder("Notepad.exe");
                Notepad.start();
                x++;
            }
            
            if (CommSplit[0].equals("hist")) {
                if (CommParam.size() < 1) {
                    System.out.println("Error :Not enough input argument(s). Please enter hist -h for help.");
                    x++;
                    
                } else {
                        if (CommParam.get(0).equals("-h")) {
                            System.out.println("hist -l [int]: Retrieve the last [int] commands entered by the user.");
                            System.out.println("hist -c: Clear the command history.");
                            x++;
                        }

                        if (CommParam.size() == 1){
                        if (CommParam.get(0).equals("-c")) {
                            System.out.println("Deleted: All command history.");
                            CommHist.clear();
                            x=x+2;
                        
}
                        }
                        
                        if (CommParam.size()>1){
                            if (CommParam.get(0).equals("-c") && CommParam.get(1).equals("-l")){
                                CommHist.clear();
                                x=x+2;
                                try{
                                   int counter = Integer.parseInt(CommParam.get(2).toString());
                                       for (int y = CommHist.size() - counter ; y<CommHist.size(); y++){
                                            if (y < 0){
                                                y=0;
                                            }
                                            try{
                                           System.out.println(CommHist.get(y));
                                            }catch(IndexOutOfBoundsException e){}
                                       }
                               }catch(NumberFormatException e){
                                   System.out.println("Please input the command in this format: hist -c -l [int]");  
                                   x=x+2;
                               
}                               
                            }
                        }
                       
                        if (CommParam.get(0).equals("-l")){
                            if (CommParam.size() <= 1){
                                System.out.println("Please input the command in this format: hist -l [int]");
                                x++;
                            }
                            if (CommParam.size() > 1){
                            x++;
                            try{
                                int counter = Integer.parseInt(CommParam.get(1).toString());
                                if (counter < 0){
                                    System.out.println("Invalid number");
                                }
                                for (int y = CommHist.size() - counter ; y<CommHist.size(); y++){
                                    if (y < 0){
                                        y=0;
                                    }
                                    try{
                                System.out.println(CommHist.get(y));
                                    }catch (IndexOutOfBoundsException e){}
                            
}
                            }catch(NumberFormatException e){
                                System.out.println("Please input the command in this format: hist -l [int]");
                                
                            }
                            if (CommParam.size() > 2){
                                if (CommParam.get(2).equals("-c")){
                                    CommHist.clear();
                                    x=2;
                                }
                                else {
                                    System.out.println("Enter input in this format to print history then clear: hist -1 [int] -c"); 
                                    x=x+2;
                                }
                            }
                            }
                        }
                    }
            }
            
            if (x == 0) 
                { System.out.println("Invalid command");}
            
            if (x != 2){CommHist.add(CommLine);}
            
        }
       
    }
}
