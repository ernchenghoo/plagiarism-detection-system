import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;


public class myshell_00000157 {

    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("Shell Version 0.01");
        
        ArrayList<String> commandHist = new ArrayList<String>();
        //Continue to loop until "exit" is entered.
        
        while (true) {
            
        	System.out.print("00000157 > ");
            commandLine = console.readLine();

            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList<String> commandParams = new ArrayList<String>();
            String[] commandSplit = commandLine.split(" ");
			
            commandHist.add(commandSplit[0]);
            	
            
		

            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
            	 //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                    
                   
                }
                
            }
           
            //Reprint the console line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
            	continue;
            }
            
            else if (commandSplit[0].equals("about")) {
            	System.out.println("Author: Jonathan Ang Rong Hwee");
            	System.out.println("Version: 0.01");
            	
            }

            //Print the current time.
            else if (commandSplit[0].equals("time")) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
            }
            
            //Print today's date.
            else if (commandSplit[0].equals("date")) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
            }
            
            //Exit the console when "exit" is entered.
            else if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }
            
            //To retrieve files and folders (or directories) from the current directory
            else if (commandSplit[0].equals("ls")) {
            	
            	File f = new File("."); // current directory
            	File[] files = f.listFiles();
            	
            	if (commandParams.size() < 1) {
            		
                	for (File file : files) {
                		if (file.isDirectory()) {
                        	System.out.print("directory: ");
                    	} else {
                        	System.out.print("file: ");
                    	}
                		System.out.println(file.getName());
                	}
                	
                } else {
                	for (File file : files) {
	                	if (commandParams.get(0).equals("-d")) {
	                		System.out.print("directory: ");
	                	}
	                	if (commandParams.get(0).equals("-f")) {
	                		System.out.print("file: ");
	                	}
	                
						System.out.println(file.getName());
	                }
                }
            }
            
            //Example of a built-in function that can be used to print the command history. 
            else if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } else {
                    //Process based on the number of parameters attached to the command.
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                    }

                    if (commandParams.get(0).equals("-c")) {
                    	commandHist.clear();
                        System.out.println("Deleted: All command history.");
                    }
                    if (commandParams.get(0).equals("-l")) {
	                    	int num;
	                    	num = Integer.parseInt(commandParams.get(1).toString()); 
	                    	System.out.println("These are your previous inputs,");
	                        
	                    	 for (int i = 0; i < num; i++){ 
	                             //print elements in commandHist   
	                                 System.out.println(commandHist.get(i));
	                    	 }
                    }
                }        	 
	            if (commandParams.size() > 2){
	                     if(commandParams.get(2).equals("-c")){ 
	                                 commandHist.clear();
	                                
	                                 System.out.println("Deleted: All command history.");
	                     }
	           }
            }
            
            else if (commandSplit[0].equals("notepad")) {
            	ProcessBuilder notepad = new ProcessBuilder();

                // Use command "notepad.exe" and open the file.
                notepad.command("notepad.exe");
                notepad.start();
            }
            else {
            		System.out.println("Invalid Input.");
            	    break;
            }
            
        	
         }
    }

	

}
