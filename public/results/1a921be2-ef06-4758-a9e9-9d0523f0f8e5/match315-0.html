import java.util.ArrayList;

public class myshell_00000130 {

    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("My Shell Version 0.01");
        
        ArrayList spaceHist = new ArrayList(); //spaceHist is commandHist
        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        boolean status = true;
        while (true) {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000130 >");

            commandLine = console.readLine();
      
            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after splitting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
       
            if (commandSplit[0].equals("about")) {
            	//Space.add(commandLine);
            	System.out.println("author: WILSON CHIAM WEI SHENG");
            	System.out.println("version: 0.01");
            }
      
            if (commandSplit[0].equals("notepad")) {
            	//Space.add(commandLine);
            	ProcessBuilder pb = new ProcessBuilder("Notepad.exe");
            	pb.start();
            }
            
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the commandline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }

            //Print the current time.
            if (commandSplit[0].equals("time")) {
            	//Space.add(commandLine);
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
            }
            
            //Print today's date.
            if (commandSplit[0].equals("date")) {
            	//Space.add(commandLine);
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
            }
            
            //Exit the console when "exit" is entered.
            if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }
            
            //To retrieve files and folders (or directories) from the current directory
            if (commandSplit[0].equals("ls")) { 
                
                //Prints the absolute path location to where the myshell project is 
                System.out.println(Paths.get(".").toAbsolutePath().normalize().toString());
                
                //Creates new file to point to the absolute path location to where the myshell project folder is
                File fol = new File(Paths.get(".").toAbsolutePath().normalize().toString());
                
                //Create Array File to store the list of files in folder
                File[] Listfiles = fol.listFiles();
                 
                   
                   if(commandParams.size()>0){ //Check whether command parameters are inserted after command ls
                        if(commandParams.get(0).equals("-f")){ //if command is ls -f
                            for (int a = 0; a < Listfiles.length; a++) { //This is to loop the array of Listfiles
                                if (Listfiles[a].isFile()) { //Check if the element exist in the array are files
                                    System.out.println(Listfiles[a].getName());
                                }
                            }
                        }
                        //if command is ls -d
                        if(commandParams.get(0).equals("-d")){ 
                             for (int a = 0; a < Listfiles.length; a++) { //loop in the array of Listfiles                              
                                if (Listfiles[a].isDirectory()){ //Check if the element eixst in array are directories
                                    System.out.println( Listfiles[a].getName());
                                }
                            }
                        }
                        
                    }else { //else print all files and directories if command is "ls"
                       for (int a = 0; a < Listfiles.length; a++) {
          
                            System.out.println(Listfiles[a].getName());
                    
                        }
                   }
            }
            
            //Example of a built-in function that can be used to print the command history. 
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } else {
                    //Process based on the number of parameters attached to the command.
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                    }
                    
                    else if(commandParams.get(0).equals("-l")) {
                    	int n;
                        
                        try{
                            n = Integer.parseInt(commandParams.get(1).toString()); //Convert the string to a integer
                            if(n<0){
                                System.out.println("Invalid number"); 
                            } 
                        } catch (NumberFormatException e){ 
                            spaceHist.add(commandLine); //Add the commandLine into spaceHist
                            System.out.println("Invalid command"); 
                            continue;
                          }
                    	
                    	int i = Integer.parseInt((String)commandParams.get(1));
                    	for(int k = 0; k<i;k++) {
                    		try {
                    		System.out.println(spaceHist.get(k));
                    		}catch (Exception e) {
                    			
                    		}
                    	}
                    }
                    
                    if (commandParams.size()>2){
                    if (commandParams.get(2).equals("-c")) {
                        spaceHist.removeAll(spaceHist);
                        System.out.println("Deleted: All command history.");
                        status = false; //command in spaceHist Array List does not included
                    }
                }
            }
        }
            
            
            if (status==true) {
            	spaceHist.add(commandLine);
            }
            
            if (!commandSplit[0].equals("about")&&!commandSplit[0].equals("hist")&&!commandSplit[0].equals("notepad")&&!commandSplit[0].equals("date")&&!commandSplit[0].equals("time")&&!commandSplit[0].equals("ls")) {
            	 System.out.println("Invalid command");
            }
            
        }
    }

}
