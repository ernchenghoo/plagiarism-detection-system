import java.io.*;
import java.text.*;
import java.util.Date;
import java.util.ArrayList;
import java.util.Collections;

public class myshell_00000063 
{
    public static void main(String[] args) throws IOException
    {
        // Use commandLine to store the command and the argument(s).
        String commandLine;

        // Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        // Just a banner
        System.out.println("My Shell Version 0.01");

        // Continue to loop until "exit" is entered.
        ArrayList<String> history = new ArrayList();
        while (true) {
            // Print the console line.
            System.out.print("00000063> ");

            // Read the input from user.
            commandLine = console.readLine();

            // Split the input into command and parameters.
            // commandParams is a list that is pre-created to store all the parameters.
            // commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");

            // Check to see if there is any parameter.
            // If the size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1)
            {
                // Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) 
                {
                    commandParams.add(commandSplit[i]);
                }
            }

            // Add the user's input into history
            history.add(commandLine);

            // Reprint the command line if user just pressed "enter" without any command.
            if (commandSplit[0].equals(""))
            {
                continue;
            }

            // If user input is "about", name and version will be displayed in the console 
            if (commandSplit[0].equals("about")) 
            {
                String name = "Teong Cai Zhi";
                double version = 0.01;
                System.out.println("author: " + name);
                System.out.println("version: " + version);
            }

            // Print current time.
            else if (commandSplit[0].equals("time")) 
            {
                Date time = new Date();
                // Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
            }

            // Print today's date.
            else if (commandSplit[0].equals("date"))
            {
                Date date = new Date();
                // Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
            }

            // Exit the console when "exit" is entered.
            else if (commandSplit[0].equals("exit"))
            {
                System.exit(0);
            }
            
            // Launch notepad
            else if(commandSplit[0].equals("notepad"))
            {
               new ProcessBuilder("notepad").start();
            }
                       
            // Retrieve files and folders (or directories) from the current directory
            else if (commandSplit[0].equals("ls"))
            {
                // Print the name of the system property
                String path = System.getProperty("user.dir");  
                File[] files = new File(path).listFiles(); 

                // If "ls" being entered, display all the files, folders or directories from the current directory
                if (commandParams.size() < 1)
                {
                    for (File file : files)
                    {
                        System.out.println(file.getName());
                    }
                } 
                // If "ls -f" is being entered, display all the files / folders in current directory
                else 
                {
                    if (commandParams.get(0).equals("-f")) 
                    {
                        for (File file : files) 
                        {
                            if (file.isFile())
                            {
                                System.out.println(file.getName());
                            }
                        }
                    }
                    // If "ls -d" is being entered, display all the directories in current directory                    
                    else if (commandParams.get(0).equals("-d")) 
                    {
                        for (File file : files) 
                        {
                            if (file.isDirectory())
                            {
                                System.out.println(file.getName());
                            }
                        }
                    }
                    // If user enter other than "ls", "ls -f" or "ls -d" display "Worng Parameter" in the console
                    else
                    {
                        System.out.println("Wrong parameter");
                    }
                }

            }

            // Built-in function that used to print the command history
            else if (commandSplit[0].equals("hist")) 
            {
                // Display error message if the user only enter "hist"
                if (commandParams.size() < 1)
                {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                }
                else 
                {
                    // Determine the input size 
                    int paramCount = commandParams.size();

                    if (paramCount == 1) 
                    { //-------------------------------------------------------------- single parameter
                        if (commandParams.get(0).equals("-c"))
                        {
                            historyDelete(history);
                        }
                        if (commandParams.get(0).equals("-h")) 
                        {
                            historyHelp();
                        }
                    } 
                    else if (paramCount == 2) 
                    { //-------------------------------------------------------------- dual parameters
                        if (commandParams.get(0).equals("-l")) 
                        {
                            historyPrint(history, commandParams.get(1).toString());
                        }
                    } 
                    else if (paramCount == 3) 
                    { //-------------------------------------------------------------- tripple parameters
                        if (commandParams.get(0).equals("-c")) 
                        {
                            historyDelete(history);
                        } 
                        else 
                        {
                            historyPrint(history, commandParams.get(1).toString());
                            historyDelete(history);
                        }
                    } 
                    else
                    {//-------------------------------------------------------------- more than 3 parameters
                        System.out.println("Too much arguments for hist command, please use -h for help");
                    }
                }
            }
            else
            {
                System.out.println("Invalid command !");
            }
        }
    }

    public static void historyDelete(ArrayList<String> history)
    {
        // Delete & clear all the previous commands
        history.clear(); 
        System.out.println("Deleted: All command history.");
    }

    public static void historyPrint(ArrayList<String> history, String num)  
    {
        try 
        {
            // Determine the number of iteration for history
            int count = Integer.parseInt(num.toString());

            // Check if the count is more than the array size 
            if (count + 1 >= history.size() + 1) 
            {
                System.out.println("Array out of bound");
            } 
            else 
            {   // Loop through & print out previous commands being entered  
                for(int i = 0; i < count; i++)
                {
                    System.out.println(history.get(i));
                }
            }            
        } catch (Exception e) 
          {
            System.out.println("Exception: " + e.toString());
          }
    }
    
    public static void historyHelp() 
    {
        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
        System.out.println("hist -c : Clear the command history.");
    }
}
