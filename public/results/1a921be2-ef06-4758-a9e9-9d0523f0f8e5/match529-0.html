import java.util.Scanner;
import java.io.*;

import java.util.ArrayList;

import java.util.Date;

import java.text.*;

public class myshell_00000102 {

    public static void main(String[] args) throws IOException {
        int change;
        
        String commandLine;

        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("author : Bashanta Pudasaini");

        System.out.println("Version 0.01");

        
          ArrayList<String> History = new ArrayList<String>();
        while (true) {

          
            System.out.print("00000102> ");

           
            commandLine = console.readLine();

            ArrayList commandParams = new ArrayList();

            String[] commandSplit = commandLine.split(" ");

            if (commandSplit.length > 1) {

                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {

                    commandParams.add(commandSplit[i]);

                }

            }
            
            

            if (commandSplit[0].equals(" ")) {

                continue;

            }

              History.add(commandLine);

            if (commandSplit[0].equals("about")) {

                System.out.println("author : Bashanta Pudasaini");

                System.out.println("Version 0.01");
                
              
            }

            
           else if (commandSplit[0].equals("time")) {

                Date time = new Date();

         
                  SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");

                    System.out.println(sdf.format(time));
                
             

            }

            
           else if (commandSplit[0].equals("date")) {

                Date date = new Date();

                
                  SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");

                    System.out.println(sdf.format(date));
                
              

            }

            //Exit the console when "exit" is entered.
          else  if (commandSplit[0].equals("exit")) {
                System.exit(0);
                
            }
            
            else if (commandSplit[0].equals("ls")) {
                if (commandParams.size() < 1){
                   
                    System.out.println("Output:");
                     File boss = new File(".");
                        File[] fail = boss.listFiles();
                        for (File file : fail) {
                            if (file.isFile() || file.isDirectory()) {
                                System.out.println(file.getName());
                            }
                            
                        }
                }
                else{
                    if (commandParams.get(0).equals("-f")){
                      
                        
                        File boss = new File(".");
                        File[] fail = boss.listFiles();
                        for (File file : fail) {
                            if (file.isFile()) {
                                System.out.println(file.getName());
                            }
                        }
                    }
                    if (commandParams.get(0).equals("-d")){
                  
                      
                        File boss = new File(".");
                        File[] fail = boss.listFiles();
                        for (File file : fail) {
                            if (file.isDirectory()) {
                                System.out.println(file.getName());
                            }
                        }
                    }
                }
            }
            
        else  if (commandSplit[0].equals("notepad")){
                ProcessBuilder notepad = new ProcessBuilder("notepad.exe");
                notepad.start();
             
            
             }
             
          
          
          else  if (commandSplit[0].equals("hist") ) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");

                }   
            
           
            
           
                else {

                    
                    if (commandParams.get(0).equals("-h")) {

                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");

                        System.out.println("hist -c : Clear the command history.");

                    }
                    
                    

                       
                if (commandParams.size() == 1){
                    if (commandParams.get(0).equals("-c")) {
                            System.out.println("command history : Deleted.");
                              History.clear();
                          
                        }
                             }
                
                if (commandParams.size()>1){
                    
                            if (commandParams.get(0).equals("-c") && commandParams.get(1).equals("-l")){
                                
                                History.clear();
                               
                                try{
                                   int count = Integer.parseInt(commandParams.get(2).toString());
                                   
                                       for (int a= History.size() - count ; a<History.size(); a++){
                                            if (a < 0){
                                                a=0;
                                            }
                                            try{
                                           System.out.println(History.get(a));
                                           
                                            }
                                            catch(IndexOutOfBoundsException e){}
                                       }
                                 }catch(NumberFormatException e){
                                     
                                   System.out.println("Input the command : hist -c -l [num]");  
                                
                               }                               
                            }
                        }
   
                            
                            

                if (commandParams.get(0).equals("-l") && commandParams.size() == 2) {
                    
                        try { 

                            change = Integer.parseInt(commandParams.get(1).toString());
                            
                            if(change > History.size()-1){
                                
                                change = History.size()-1;
                            }
                            if (change >= 0 ){ 
                                for(int i = History.size()-2; i >= History.size()-change-1; i--){
                                    
                                    System.out.println(History.get(i));
                                } 
                            }
                            
                            else{
                                
                                System.out.println("Invalid Number !");
                            }
                            
                        }catch (Exception e){

                            System.out.println("Invalid Number !");
                        }

                    }
                  

                  
                }
                if (commandParams.size() == 3) {
                    
                    if (commandParams.get(0).equals("-l") && commandParams.get(2).equals("-c")) {
                        
                        try {
                            
                            change = Integer.parseInt((String) commandParams.get(1));
                            
                                if(change > History.size()-1){
                                    
                                     change = History.size()-1;
                                     
                            }
                                

                            if (change >= 0 ) {
                                
                                for(int i = History.size()-2; i >= History.size()-change-1; i--){
                                    
                                    System.out.println(History.get(i));
                                }

                            }
                                 else {
                                
                                   System.out.println("Invalid Number");
                            }
                            
                            
                            if (commandParams.get(2).equals("-c") && change > 0) {
                                History.clear();
                                System.out.println("Command History Deleted");
                            }

                        }catch (Exception e){
                            
                            System.out.println("Invalid Number!");
         }

        }
         
        }
        } 
          else{System.out.println("Invalid comand!!");}
         
    }
    }
    }
