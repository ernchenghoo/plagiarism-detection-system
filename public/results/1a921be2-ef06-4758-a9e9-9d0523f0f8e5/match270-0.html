import java.io.*;
import java.util.*;
import java.text.*;


public class myshell_00000141 {
    
    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        ArrayList history = new ArrayList();

        //Just a banner
        System.out.println("My Shell Version 0.01");
        String name = new String("Leong Jia Lin");
        double version = 0.01;
        
        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            boolean found = false;
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000141> ");

            //Read the input from user.
            commandLine = console.readLine();
            
            //history is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            String commandSplit[] = commandLine.split(" ");
            history.add(commandLine);       
            
            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }
            //Print about
            if(commandSplit[0].equals("about") && commandSplit.length == 1){
                System.out.println("author: " + name);
                System.out.println("version: " + version);
                found = true;
            }
            //Open notepad application
            if(commandSplit[0].equals("notepad") && commandSplit.length == 1){
                ProcessBuilder notepad = new ProcessBuilder(commandSplit[0]);
                notepad.start();
                found = true;
            }
            
            //Print the current time.
            if(commandSplit[0].equals("time") && commandSplit.length == 1) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                found = true;
            }
                        
            //Print today's date.
            if (commandSplit[0].equals("date") && commandSplit.length == 1) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                found = true;
            }
            
            //Exit the console when "exit" is entered.
            if (commandSplit[0].equals("exit") && commandSplit.length == 1) {
                System.exit(0);
                found = true;
            }
            
            //To retrieve files and folders (or directories) from the current directory
            if (commandSplit[0].equals("ls")) {
                File directory = new File("D:\\D.Document");
                File[] fList = directory.listFiles();
                if(commandSplit.length == 2){
                    if(commandSplit[1].equals("-f")){
                       for(File file : fList){
                           if(file.isFile()){ 
                               System.out.println(file.getName());
                           }
                       }
                       found = true;
                    }
                    if(commandSplit[1].equals("-d")){
                        for(File file : fList){
                            if(file.isDirectory()){
                                System.out.println(file.getName());
                            }
                        }
                       found = true; 
                    }    
                }
                else{
                    for(File file : fList){
                        System.out.println(file.getName());
                    } 
                    found = true;
                }                                  
            }
            
            //Example of a built-in function that can be used to print the command history. 
            if (commandSplit[0].equals("hist")) {
                if (history.size() == 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    found = true;
                } else{
                    for(int i = 1; i< commandSplit.length; i++){
                        //Process based on the number of parameters attached to the command.
                        if (commandSplit[i].equals("-h")) {
                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                            System.out.println("hist -c : Clear the command history.");
                            found = true;
                        }

                        if (commandSplit[i].equals("-c")) {
                            System.out.println("Deleted: All command history.");
                            history.removeAll(history);
                            found = true;
                        }
                        if (commandSplit[i].equals("-l")){
                            if(commandSplit.length < 3){
                                for(int count = 0; count < history.size() - 1; count++){
                                    System.out.println(history.get(count));
                                }
                            }
                            else {
                                //input value > history size
                                if(history.size()<Integer.parseInt(commandSplit[i + 1])){
                                    for(int count = 0; count < history.size() - 1; count++){
                                        System.out.println(history.get(count));
                                    }
                                }
                                //negative value
                                else if(Integer.parseInt(commandSplit[i + 1]) < 0){
                                    System.out.println("Invalid integer");
                                }
                                //input value < history size
                                else{
                                    for(int count = history.size() - 1 - Integer.parseInt(commandSplit[i + 1]); count < history.size()-1; count++){
                                        System.out.println(history.get(count));
                                    }
                                }
                            }
                            found = true;
                        }
                    }                    
                }         
            }
            if(found == false){
                System.out.println("Invalid command");
            }
        }
    }
}
