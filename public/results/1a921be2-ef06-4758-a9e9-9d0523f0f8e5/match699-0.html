import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class Myshell_00000126 {

    private static void printHistory(int input, ArrayList commandHistory){
        if(commandHistory != null){
            if(input < 0){
                System.out.println("Invalid Number!");
            }
            else {
                try {
                    for (int i = 0; i < input; i++) {
                        System.out.println(commandHistory.get(i));

                        //Prevent printing current command
                        if (i == commandHistory.size() - 1) {
                            // System.out.println("End of History");
                            break;
                        }
                    }
                } catch (Exception e) {
                    System.out.println("Invalid number or empty history");
                }
            }
        }
        else {
            System.out.println("No history yet!");
        }
    }

    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("Ngu Chee Hong (00000126) CSC2104 Mini Assignment Shell");

        ArrayList<String> commandHistory = new ArrayList<String>();

        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000126> ");

            //Read the input from user.
            commandLine = console.readLine();

            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");

            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }

            //Display author and version number
            else if (commandSplit[0].equals("about")) {
                System.out.println("author: Ngu Chee Hong");
                System.out.println("version: 0.01");
                commandHistory.add(commandLine);
            }

            //Print the current time.
            else if (commandSplit[0].equals("time")) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                commandHistory.add(commandLine);
            }

            //Print today's date.
            else if (commandSplit[0].equals("date")) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                commandHistory.add(commandLine);
            }

            //Exit the console when "exit" is entered.
            else if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }

            //To retrieve files and folders (or directories) from the current directory
            else if (commandSplit[0].equals("ls")) {
                if (commandParams.size() < 1) {
                    System.out.println("Listing everything in current directory.");
                    File file;
                    File[] paths;
                    try {
                        file = new File(".");
                        paths = file.listFiles();
                        for (File path : paths) {
                            System.out.println(path);
                        }
                    } catch (Exception e) {
                        System.err.println("Things went wrong: " + e.getMessage());
                    }
                }
                else {
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("ls -f : list files only in current directory");
                        System.out.println("ls -d : list folders only in current directory");
                        System.out.println("ls : list both files and folders in current directory");
                    }
                    else if (commandParams.get(0).equals("-f")) {
                        System.out.println("Listing all files in current directory.");
                        File file;
                        File[] paths;
                        try {
                            file = new File(".");
                            paths = file.listFiles();
                            for (File path : paths) {
                                if(!path.isDirectory()) {
                                    System.out.println(path);
                                }
                            }
                        } catch (Exception e) {
                            System.err.println("Things went wrong: " + e.getMessage());
                        }
                    }
                    else if (commandParams.get(0).equals("-d")) {
                        System.out.println("Listing all folders in current directory.");
                        File file;
                        File[] paths;
                        try {
                            file = new File(".");
                            paths = file.listFiles();
                            for (File path : paths) {
                                if(path.isDirectory()) {
                                    System.out.println(path);
                                }
                            }
                        } catch (Exception e) {
                            System.err.println("Things went wrong: " + e.getMessage());
                        }
                    }
                    else {
                        System.out.println("Wrong argument!");
                    }
                }
                commandHistory.add(commandLine);
            }

            //opens gedit (allows argument to state filename)
            else if (commandSplit[0].equals("gedit") || commandSplit[0].equals("notepad")) {
                if (commandParams.size() == 1){
                    String filename = commandParams.get(0).toString();
                    ProcessBuilder pb = new ProcessBuilder("bash", "-c", "gedit " + filename);
                    pb.redirectErrorStream(true);
                    pb.start();
                }
                else {
                    System.out.println("You can also add an argument to state the filename!");
                    ProcessBuilder pb = new ProcessBuilder("bash", "-c", "gedit");
                    pb.redirectErrorStream(true);
                    pb.start();
                }
                commandHistory.add(commandLine);
            }

            //Example of a built-in function that can be used to print the command history.
            else if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                }
                else if (commandParams.size() == 1){
                    //Process based on the number of parameters attached to the command.
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                        commandHistory.add(commandLine);
                    }
                    else if (commandParams.get(0).equals("-c")) {
                        try{
                            commandHistory.clear();
                            System.out.println("Deleting history...");
                            System.out.println("Deleted: All command history.");
                        }
                        catch (Exception e) {
                            System.out.println("No history!");
                        }
                    }
                    else if (commandParams.get(0).equals("-l")) {
                        printHistory(commandHistory.size(), commandHistory);
                    }
                }
                else if (commandParams.size() == 2 && commandParams.get(0).equals("-l")) {
                    String coman = (String) commandParams.get(1);
                    int input = Integer.parseInt(coman);
                    printHistory(input, commandHistory);
                    commandHistory.add(commandLine);
                }
                else if (commandParams.size() == 3 && commandParams.get(0).equals("-l") && commandParams.get(2).equals("-c")) {
                    String coman = (String) commandParams.get(1);
                    int input = Integer.parseInt(coman);
                    printHistory(input, commandHistory);
                    try{
                        commandHistory.clear();
                        System.out.println("Deleting history...");
                        System.out.println("Deleted: All command history.");
                    }
                    catch (Exception e) {
                        System.out.println("No history!");
                    }
                }
                else if (commandParams.size() == 3 && commandParams.get(0).equals("-c") && commandParams.get(1).equals("-l")){
                    String coman = (String) commandParams.get(2);
                    int input = Integer.parseInt(coman);
                    try{
                        commandHistory.clear();
                        System.out.println("Deleting history...");
                        System.out.println("Deleted: All command history.");
                    }
                    catch (Exception e) {
                        System.out.println("No history!");
                    }
                    printHistory(input, commandHistory);
                }
                else {
                    System.out.println("Bad arguments!");
                }
            }

            else {
                System.out.println("Invalid command!");
                commandHistory.add(commandLine);
                continue;
            }
            ////////////////////////////////////
        }
    }
}
