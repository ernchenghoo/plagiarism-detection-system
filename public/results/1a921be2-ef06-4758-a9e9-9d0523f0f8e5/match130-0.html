import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;


public class myshell_00000124 {
    
        public static void main(String[] args) throws IOException{
        String commandLine;
        
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        
        System.out.println("My Shell Version 0.01");
        ArrayList commandInput = new ArrayList();
        
        while(true){
            System.out.print("00000124> ");
            
            commandLine = console.readLine();
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            
            if(commandSplit.length > 1){
                for(int i=1; i<commandSplit.length; i++){
                    commandParams.add(commandSplit[i]);
                }
            }
            
            if (commandSplit[0].equals("")) {
                continue;
            }
            
            else if (commandSplit[0].equals("about")){
                System.out.println("author: Dephanie Neoh Shin Yee");
                System.out.println("version: 0.01");
                commandInput.add("about");
            }
            
            else if (commandSplit[0].equals("time")){
                Date time = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat ("hh:mm:ss a");
                System.out.println(sdf.format(time));
                commandInput.add("time");
            }
            
            else if (commandSplit[0].equals("date")){
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                commandInput.add("date");
            }
            
            else if(commandSplit[0].equals("exit")){
                System.exit(0);
            }
            
            else if(commandSplit[0].equals("notepad")){
                ProcessBuilder pb = new ProcessBuilder("Notepad.exe");
                pb.start();
                commandInput.add("notepad");
            }
      
            else if(commandSplit[0].equals("ls")){
                if(commandParams.size() < 1){
                    File file = new File(".");
                    String[] files = file.list();
                    for(String string : files){
                        System.out.println(string);
                    }
                    commandInput.add("ls");
                }
                else if(commandParams.size()< 2){
                    if(commandParams.get(0).equals("-f")){
                        File[] files = new File(".").listFiles();
                        for(File file : files){
                            if(file.isFile()){
                                System.out.println(file.getName());
                            }
                        }
                        commandInput.add("ls -f");
                    }
                    else if(commandParams.get(0).equals("-d")){
                        File[] directory = new File(".").listFiles();
                        for(File file : directory){
                            if(file.isDirectory()){
                                System.out.println(file.getName());
                            }
                        }
                        commandInput.add("ls -d");
                        
                    }
                    else{
                    System.out.println("Invalid command!");
                    }
                }
                else{
                    System.out.println("Invalid command!");
                }     
            }
            
            
            else if(commandSplit[0].equals("hist")){
                if(commandParams.size() < 1){
                    System.out.println("Error: Not enough input argument(s). Please enter hist -h for help.");          
                }
                else if(commandParams.size() < 2){
                    if(commandParams.get(0).equals("-h")){
                        System.out.println("hist -1 [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                        commandInput.add("hist -h");
                    }
                    
                    else if(commandParams.get(0).equals("-c")){
                        commandInput.clear();
                        System.out.println("Deleted: All command history."); 
                    }
                    else{
                        System.out.println("Invalid command!");
                    }
                }
                else if (commandParams.size() < 3){
                    if(commandParams.get(0).equals("-l") && commandInput.size()>0){
                        try{
                            String s = commandSplit[2];
                            Integer num = Integer.valueOf(s);
                            for(int i=(commandInput.size()-num); i < commandInput.size(); i++){
                                System.out.println(commandInput.get(i));
                            }
                            String input = new String("hist -l " + num);
                            commandInput.add(input);
                        }    
                        catch(NumberFormatException e){
                            System.out.println("Invalid number!");
                        }
                        catch(ArrayIndexOutOfBoundsException e){
                            System.out.println("Error! Please enter a lower number.");
                        }
                    }
                    else if(commandInput.size() == 0){
                        try{
                            String s = commandSplit[2];
                            Integer num = Integer.valueOf(s);
                            String input = new String("hist -l " + num);
                            commandInput.add(input);
                        }    
                        catch(NumberFormatException e){
                            System.out.println("Invalid number!");
                        }
                        catch(ArrayIndexOutOfBoundsException e){
                            continue;
                        }
                        continue;
                    }
                    else{
                        System.out.println("Invalid command!");
                    }
                }
                else if (commandParams.size() < 4){
                    if(commandParams.get(0).equals("-l") && commandInput.size()>0 && commandParams.get(2).equals("-c")){
                        try{
                            String s = commandSplit[2];
                            Integer num = Integer.valueOf(s);
                            for(int i=(commandInput.size()-num); i < commandInput.size(); i++){
                                System.out.println(commandInput.get(i));
                            }
                            commandInput.clear();
                            System.out.println("Deleted: All command history.");
                            String input = new String("hist -l " + num + " -c");
                            commandInput.add(input);
                        }    
                        catch(NumberFormatException e){
                            System.out.println("Invalid number!");
                        }
                        catch(ArrayIndexOutOfBoundsException e){
                            System.out.println("Error! Please enter a lower number.");
                        }
                    }
                    else if(commandParams.get(0).equals("-c")&& commandInput.size()>0 && commandParams.get(1).equals("-l")){
                        commandInput.clear();
                        System.out.println("Deleted: All command history.");
                        try{
                            String s = commandSplit[3];
                            Integer num = Integer.valueOf(s);
                            String input = new String("hist -l " + num + " -c");
                            commandInput.add(input);
                        }    
                        catch(NumberFormatException e){
                            System.out.println("Invalid number!");
                        }
                        catch(ArrayIndexOutOfBoundsException e){
                            commandInput.clear();
                            System.out.println("Deleted: All command history.");
                        }
                    }
                    else if(commandInput.size() == 0){
                        continue;
                    }
                    else{
                        System.out.println("Invalid command!");
                    }
                }
                else{
                    System.out.println("Invalid command!");
                }
                
                     
            }
            else{
                    System.out.println("Invalid command!");
            }
        }           
    }
}
