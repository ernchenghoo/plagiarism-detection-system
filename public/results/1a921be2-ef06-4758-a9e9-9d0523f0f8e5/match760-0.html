import java.io.BufferedReader;
import java.io.File;
import java.io.IOException;
import java.io.InputStreamReader;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Date;
import java.util.Iterator;
import java.util.ListIterator;


public class myshell_00000081 {
    
    public static void main(String[] args) throws IOException {
        String cmdLine;
        ArrayList cmdHistory = new ArrayList(); //storing past commands
        Iterator it = cmdHistory.iterator();
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        System.out.println("My Shell");
        boolean exit = false;
        while(!exit){
            System.out.print("00000081> ");
            cmdLine = br.readLine();
            if(!cmdLine.equals(""))
                cmdHistory.add(cmdLine);
            //splitting input into command and parameters
            ArrayList cmdParams = new ArrayList();
            String[] cmdSplit = cmdLine.split(" ");

            //checks if user entered parameters inside command
            if(cmdSplit.length > 1){
                for (int i = 1; i < cmdSplit.length; i++)
                    cmdParams.add(cmdSplit[i]);
                
                //removes extra spaces
                for (int i = 0; i < cmdParams.size(); i++){
                    if (cmdParams.get(i).equals("")){   
                        cmdParams.remove(i);
                        i--;
                    }
                }
                    
            }
            
            switch(cmdSplit[0]){
                case "":
                    continue;
                case "exit":
                    System.exit(0);
                    break;
                case "about":
                    System.out.println("Author: Tan Yi Yang");
                    System.out.println("Id: 00000081");
                    System.out.println("Version: 0.01");
                    break;
                case "time":
                    Date time = new Date();
                    SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                    System.out.println(sdf.format(time));
                    break;
                case "date":
                    Date date = new Date();
                    SimpleDateFormat dSdf = new SimpleDateFormat("E dd/MM/yyyy");
                    System.out.println(dSdf.format(date));
                    break;
                case "notepad":
                    ProcessBuilder pb = new ProcessBuilder("Notepad.exe");
                    pb.start();
                    break;
                case "ls":
                    File folder = new File(System.getProperty("user.dir")); 
                    File[] fileList = folder.listFiles();
                    
                    //listing all files and directory if no parameters are entered
                    if (cmdParams.size() < 1){
                        for(int i = 0; i < fileList.length; i++){
                            if(fileList[i].isFile())
                                System.out.println("\t"+fileList[i].getName());
                            else
                                System.out.println("<DIR>\t"+fileList[i].getName());
                        }
                    }
                    
                    //provide details depending on parameters entered
                    else {
                        for(int i = 0; i < cmdParams.size(); i++){
                            if (cmdParams.get(i).equals("-f")){
                                for(int j = 0; j < fileList.length; j++){
                                    if(fileList[j].isFile()){
                                        System.out.println("\t" + fileList[j].getName());
                                    }
                                }
                            }
                            else if (cmdParams.get(i).equals("-d")){
                                for(int j = 0; j < fileList.length; j++){
                                    if(fileList[j].isDirectory()){
                                        System.out.println("<DIR>\t" + fileList[j].getName());
                                    }
                                }
                            }
                            else
                                System.out.println("Invalid parameter: " + cmdParams.get(i));
                        }
                    }
                    break;
                case "hist":
                    if (cmdParams.size() < 1)
                        System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");

                    else {
                        for(int i = 0; i<cmdParams.size();i++){
                            if (cmdParams.get(i).equals("-h")){
                                System.out.println("hist -1 [int] : Retrieve the last [int] commands entered by the user");//work on it
                                System.out.println("hist -c : Clear the command history.");
                            }
                            else if (cmdParams.get(i).equals("-l")){
                                it = cmdHistory.iterator(); //updates iterator
                                i++; //to check next value after '-l'
                                if (cmdParams.size()<i+1){
                                    System.out.println("Missing parameter");
                                    continue;
                                }
                                
                                int num = cmdHistory.size() - Integer.valueOf((String)cmdParams.get(i)) - 1; // num : the index of which command it should start printing from, -1 to ignore last command
                                int count = 0;

                                if (num < 0)
                                    System.out.println("Invalid parameter");
                                else{
                                    while(it.hasNext()){
                                        if (count >= num && count < cmdHistory.size()-1)
                                            System.out.println(it.next());
                                        else
                                            it.next();
                                        count++;
                                    }
                                }
                            }
                            else if (cmdParams.get(i).equals("-c")){
                                cmdHistory.clear();
                                System.out.println("Deleted: All command history.");
                            }
                            else
                                System.out.println("Invalid parameter: " + cmdParams.get(i));
                        }
                    }
                    break;
                default:
                    System.out.println("Invalid command");
            }
        }
    }
}
