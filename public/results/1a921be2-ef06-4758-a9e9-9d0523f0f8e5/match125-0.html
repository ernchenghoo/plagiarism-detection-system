import java.io.*;
import java.util.ArrayList;

public class myshell_00000022 {

    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;
        ArrayList x = new ArrayList(); 
        int c = -1; //count the input which entered
        String n;  //for replacing the value from commandParams(1) and subtract it to int c to find the last c command
        //Read and buffer the commands.
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("Shell Version 0.01");

        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {  
            //Print the console.
            System.out.print("\n00000022> ");

            //Read the input.
            commandLine = console.readLine();

            //Split the input into command and parameters.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
           
            
            //#1 Displaying author and the version
            if (commandSplit[0].equals("about")) {
            	System.out.println("Author: Naohiro");
            	System.out.println("Version: 0.01");
            }
            
            x.add(commandLine);

            //Check whether there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the console line if user just pressed "enter".
            if (commandSplit[0].equals("")) {
                continue;
            }
            
            //#5 to launch notepad.
            if (commandSplit[0].equals("notepad")) {
            	ProcessBuilder np = new ProcessBuilder("notepad");
            	np.start();
            }

            //Exit the console when "exit" is entered.
            if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }
            c = c + 1; //to count how many command entered.
            
            //Built-in function that can be used to trace the command history.            
            if (commandSplit[0].equals("hist")) 
            {
       
                if (commandParams.size() < 1) 
                {
                    System.out.println("Error : Not enough input argument. Please enter hist -h for help.");
                } 
                
                //#2 if hist -l x(integer) 
                if (commandSplit[1].equals("-1") && commandParams.size() > 1)
                {
                	n = (String) commandParams.get(1);
                 	int nInteger = Integer.parseInt(n); //change string to int          	  
                	System.out.println(x.get(c - nInteger)); 
                	
                
                }
                
                //#4 if hist -1 x -c entered.
                if (commandSplit[1].equals("-1") && commandParams.size() > 2 && commandSplit[2].equals("-c"))
                {
                	n = (String) commandParams.get(1);
                 	int nInteger = Integer.parseInt(n); //change string to int          	  
                	System.out.println(x.get(c - nInteger)); 
                	
                	System.out.println("Command history is deleted.");

                        System.out.printf("Deleted " + c + " input..."); 

                        System.out.printf("history: " + x); //show deleted command
                        c = -1; //clear number of command entered
                        x.clear(); //clear history
                }
                	          
                
                else 
                {
                    //Process based on the number of parameters attached to the command.
                    if (commandParams.get(0).equals("-h")) 
                    {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                    }

                    
                    //#3 delete history
                    if (commandParams.get(0).equals("-c")) 
                    {
                        System.out.println("Command history has been deleted.");

                        System.out.printf("Deleted " + c + " input..."); 

                        System.out.printf("history: " + x); //show deleted command
                        c = -1; //clear number of command entered
                        x.clear(); //clear history
                        
                    }
                }//end of else

	            
            }//end of if "hist"

          
            //#6 invalid command.
            else
            {
            	System.out.println("Invalid command");
            }
            
    }//end
        
}

	
}
