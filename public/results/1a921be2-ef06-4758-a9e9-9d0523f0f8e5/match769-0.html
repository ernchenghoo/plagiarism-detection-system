import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;


public class myshell_00000082 {
    

    public static void main(String[] args) throws IOException {

        String commandLine;

   
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

       
        System.out.println("My Shell Version 0.01");
        
        
        ArrayList <String> al = new ArrayList();
        
        while (true) {
            int commandValid = 0;
          
            
            System.out.print("00000082 > ");

            
            commandLine = console.readLine();
            
            
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            
            
            if(commandSplit.length > 1){
                for(int i = 1; i < commandSplit.length; i++){
                    commandParams.add(commandSplit[i]);
                }
            }
            
            
            if(commandSplit[0].equals("")){
                continue;
            }
            
            
            if(commandSplit[0].equals("about") && commandSplit.length == 1){
                System.out.println("author: Khoo Ter Kuan,Sean");
                System.out.println("version: 0.01");
                al.add(commandLine);
                commandValid = 1;  
            }
             
            
            if(commandSplit[0].equals("time") && commandSplit.length == 1){
                Date time = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                al.add(commandLine);
                commandValid = 1;  
            }
            
            
            if(commandSplit[0].equals("date")&& commandSplit.length == 1){
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                al.add(commandLine);
                commandValid = 1;  
            }
            
            
            if(commandSplit[0].equals("exit")&& commandSplit.length == 1){
                System.exit(0);
                commandValid = 1;  
            }
            
            
            
            if(commandSplit[0].equals("ls") && commandSplit.length == 2){
                File dir = new File(".");
                File[] filesList = dir.listFiles();
                
                if(commandSplit.length == 1){
                    for(File file : filesList){
                        System.out.println(file.getName());
                    }
                }   
                    else{
                        for(File file : filesList){
                            if(file.isFile() && commandSplit[1].equals("-f")){
                                System.out.println(file.getName());
                            }

                            if(file.isDirectory() && commandSplit[1].equals("-d")){
                                System.out.println(file.getName());
                            }
                        }          
                    }   
            al.add(commandLine);
            commandValid = 1;  
            }
       
            
            if(commandSplit[0].equals("notepad")&& commandSplit.length == 1){
                ProcessBuilder pb = new ProcessBuilder("Notepad.exe");
                File file = new File("Notepad.txt");
                pb.start();
                al.add(commandLine);
                commandValid = 1;  
            }
           
            
            if (commandSplit[0].equals("hist")){
                if (commandParams.size() < 1){
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                }
                
                else{
                    
                    if(commandParams.get(0).equals("-h")){
                                System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                                System.out.println("hist -c : Clear the command history.");
                                
                                
                    }

                                if(commandParams.get(0).equals("-l") && !al.isEmpty()){
                                    int x = Integer.parseInt((String)commandParams.get(1));
                                        if( x<0 || x > al.size()){
                                           System.out.println("Invalid number");
                                           al.add(commandLine);//add invalid number into al
                                           continue;
                                        }
                                        for(int i = al.size()-x; i<al.size(); i++){
                                                System.out.println(al.get(i));
                                            }
                                }


                                    if(commandParams.get(0).equals("-c") && commandParams.size()<2){
                                        System.out.println("Deleted: All command history.");
                                        al.clear();
                                        
                                    }


                                        if(commandParams.size() == 3){
                                            if(commandParams.get(0).equals("-c") ||commandParams.get(2).equals("-c") ){
                                                System.out.println("Deleted: All command history.");
                                                al.clear();
                                            }

                                            if(commandParams.get(1).equals("-l") && !al.isEmpty()) {
                                                int x = Integer.parseInt((String) commandParams.get(2));
                                                    if( x<0 || x > al.size()){
                                                        System.out.println("Invalid number");
                                                        al.add(commandLine);//add invalid number into al
                                                        continue;
                                                    }
                                                        for(int i = al.size()-x; i < al.size();i++){
                                                            System.out.println(al.get(i));
                                                        }
                                            }
                                         
                                        }
            
                }   
             al.add(commandLine);
             commandValid = 1;      
             }
            
            if(commandValid == 0){
                System.out.println("Invalid command!");
                al.add(commandLine);
            }
        }
    }
}
