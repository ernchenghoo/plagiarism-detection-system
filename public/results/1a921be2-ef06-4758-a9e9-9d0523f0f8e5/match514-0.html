import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000127 {
 public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        //comm is a list that is used to store user inputs
        ArrayList comm = new ArrayList();

        //Just a banner
        System.out.println("My Shell Version 0.01");

        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000127> ");

            //Read the input from user.
            commandLine = console.readLine();

            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            //validcomm is a list that is used to store valid commands
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            ArrayList<String> validcomm = new ArrayList<>();
            
            
            //Add valid commands to Array
            validcomm.add("date");
            validcomm.add("time");
            validcomm.add("ls");
            validcomm.add("hist");
            validcomm.add("notepad");
            validcomm.add("about");
            validcomm.add("exit");
            
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }

            //Prints info
            if (commandSplit[0].equals("about")) {
                comm.add(commandLine);
                System.out.println("author: William Yap Wai Nam\nversion: 0.01");
            }
            //Print the current time.
            if (commandSplit[0].equals("time")) {
                comm.add(commandLine);
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
            }

            //Print today's date.
            if (commandSplit[0].equals("date")) {
                comm.add(commandLine);
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
            }

            //Exit the console when "exit" is entered.
            if (commandSplit[0].equals("exit")) {
                comm.clear();
                System.exit(0);
            }

            //To retrieve files and folders (or directories) from the current directory
            if (commandSplit[0].equals("ls")) {
                comm.add(commandLine);
                if (commandParams.isEmpty()) {
                    printFiles();
                    printDirectory();
                } else {
                    //List only files
                    if (commandParams.get(0).equals("-f")) {
                        comm.add(commandLine);
                        printFiles();
                    }
                    
                    //List only folders
                    if (commandParams.get(0).equals("-d")) {
                        comm.add(commandLine);
                        printDirectory();
                    }
                }
            }

            //Example of a built-in function that can be used to print the command history. 
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                  
                    
                } else {
                        //Display help messages
                        if (commandParams.get(0).equals("-h")) {
                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                            System.out.println("hist -c : Clear the command history.");
                        }
                        
                        //Clear command history
                        if(commandParams.get(0).equals("-c")) {
                            System.out.println("Deleted: All command history");
                            comm.clear();
                        }
                        
                        //Show command history
                        else if(commandParams.get(0).equals("-l")) {
                            int param = Integer.parseInt((String) commandParams.get(1));
                            
                            if(param > 0) {
                                if(comm.isEmpty()) {
                                    continue;
                                }
                                
                                if(comm.size()<param) {
                                    System.out.println("Invalid");
                                }
                                else for(int n = 0; n < param; n++) {
                                    System.out.println(comm.get(n));
                                }
                            }
                            if(param < 0){
                                if(comm.isEmpty()) {
                                    continue;
                                }
                                else {
                                    System.out.println("Invalid!");
                                }
                            }
                        }
                        
                        if(commandParams.size() > 2) {
                            if(commandParams.get(2).equals("-c")) {
                                System.out.println("Deleted: All command history");
                                comm.clear();
                            }
                        }
                }
            }
            
            //Launch notepad
            if(commandSplit[0].equals("notepad")){
                comm.add(commandLine);
                Runtime rs = Runtime.getRuntime();
                rs.exec("notepad");
            }
            
            //Check input validity
            boolean validity = false;
            for(int k = 0; k < validcomm.size(); k++) {
                if(commandSplit[0].equals(validcomm.get(k))) {
                    validity = true;
                }
            }
            
            //Prints false message if invalid
            if(validity == false) {
                System.out.println("Invalid command!");
            }
        }
       
    }
        
    public static void printFiles() {
        File file = new File(".");
        File[] filesList = file.listFiles();

        for (File f : filesList) {
            if (f.isFile()) {
                System.out.println(f.getName());
            }
        }
    }

    public static void printDirectory() {
        File file = new File(".");
        File[] filesList = file.listFiles();

        for (File f : filesList) {
            if (f.isDirectory()) {
                System.out.println(f.getName());
            }
        }
    }
}
