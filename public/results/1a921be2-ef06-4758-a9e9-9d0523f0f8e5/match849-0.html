import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000092 {
    
    public static void main(String[] args) throws IOException {
        
        String commandLine;
        
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        
        System.out.println("My Shell Version 0.01");
        
        ArrayList cmdHist = new ArrayList();
        
        while(true) {
            int cmd = 0;
            System.out.print("00000092> ");
            
            commandLine = console.readLine();
             
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            if (commandSplit.length > 1) {
                
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            } 
            if (commandSplit[0].equals("")) {
                continue;
            }
            
            
            if (commandSplit[0].equals("about")){
                System.out.println("author: Wan Ray Chi");
                System.out.println("Version: 0.01");
                cmd++;
            }
            //Q num 1
            
            if (commandSplit[0].equals("time")) {
                Date time = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                cmd++;
            }
            
            if (commandSplit[0].equals("date")) {
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                cmd++;
            }
            
            if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }
            
            
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    cmd++;
                } else {
                        if (commandParams.get(0).equals("-h")) {
                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                            System.out.println("hist -c : Clear the command history.");
                            cmd++;
                        }

                        if (commandParams.size() == 1){
                        if (commandParams.get(0).equals("-c")) {
                            System.out.println("Deleted: All command history.");
                            cmdHist.clear();
                            cmd=cmd+2;
                        }
                        }
                        
                        if (commandParams.size()>1){
                            if (commandParams.get(0).equals("-c") && commandParams.get(1).equals("-l")){
                                cmdHist.clear();
                                cmd=cmd+2;
                                try{
                                   int count = Integer.parseInt(commandParams.get(2).toString());
                                       for (int x = cmdHist.size() - count ; x<cmdHist.size(); x++){
                                            if (x < 0){
                                                x=0;
                                            }
                                            try{
                                                System.out.println(cmdHist.get(x));
                                            }catch(IndexOutOfBoundsException e){
                                            }
                                       }
                                }catch(NumberFormatException e){
                                    System.out.println("Number Format Error. Input: hist -c -l [int]");  
                                   cmd=cmd+2;
                               }                               
                            }
                        }
                        if (commandParams.get(0).equals("-l")){
                            if (commandParams.size() <= 1){
                                System.out.println("Missing [int] input: hist -l [int]");
                                cmd++;
                            }
                                if (commandParams.size() > 1){ 
                                    cmd++;
                                    try{
                                        int count = Integer.parseInt(commandParams.get(1).toString());
                                        if (count < 0){
                                        System.out.println("Error: Invalid Number");
                                        }
                                            for (int x = cmdHist.size() - count ; x<cmdHist.size(); x++){
                                                if (x < 0){
                                                    x=0;
                                                }
                                                try{
                                                    System.out.println(cmdHist.get(x));
                                                }catch (IndexOutOfBoundsException e){}
                                            }
                                    }catch(NumberFormatException e){
                                        System.out.println("Number Format Error. Input: hist -l [int] -c");
                                    }
                                    
                            if (commandParams.size() > 2){
                                if (commandParams.get(2).equals("-c")){
                                    cmdHist.clear();
                                    cmd=cmd+2;
                                }
                                else {
                                    System.out.println("Enter input in this format to print history then clear: hist -1 [int] -c"); 
                                    cmd=cmd+2;
                                }
                            }
                            }
                        }
                }
           }
            //Q num 2 - 4
            
            if (commandSplit[0].equals("ls")) {
                if (commandParams.size() < 1){
                    cmd++;
                    System.out.println("List:");
                    File f = new File(".");
                        File[] floc = f.listFiles();
                        for(File file:floc) {
                            if (file.isFile() || file.isDirectory()) {
                                System.out.println(file.getName());
                            }
                        }
                }
                else { 
                    if (commandParams.get(0).equals("-f")){
                        cmd++;
                        System.out.println("Listed File:");
                        File f = new File(".");
                        File[] floc = f.listFiles();
                        for (File file : floc) {
                            if (file.isFile()) {
                                System.out.println(file.getName());
                            }
                        }
                    }
                
                else if (commandParams.get(0).equals("-d")){
                        cmd++;
                        System.out.println("Directory List:");
                        File f = new File(".");
                        File[] floc = f.listFiles();
                        for (File file : floc) {
                            if (file.isDirectory()) {
                                System.out.println(file.getName());
                            }
                        }
                    }
                }
            }
            //Q num 5
            
           if (commandSplit[0].equals("notepad")){
                ProcessBuilder p = new ProcessBuilder("notepad.exe");
                p.start();
                cmd++;
            } // Q num 6
           
           if (cmd == 0) 
               { System.out.println("Invalid command");} 
                // Q num 7
           
           else if (cmd != 2){cmdHist.add(commandLine);}
        }
    }
}
