package myshell_00000119;
/**

 */
import java.io.*;
import java.util.ArrayList;
import  java.util.Date;
import java.text.*;


public class Myshell_00000119 {

    private static Object sdf;

    public static void main(String[] args) throws IOException
    {
        //commandLine to store the command and the argument(s).
        String commandLine;
        //commandOutput is a list of previous output entered by the user.
        ArrayList commandOutput = new ArrayList();
        //variable to store parameter in integer type.
        int number = 0;
        //Information output when input "about" command.
        String name = "MOHAMMED";
        double version = 0.01;
        
        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        
        //Information
        System.out.println("My Shell Version 0.01");
        
        //Continue to loop until "exit"(false) is entered.
        while(true)
        {
            //Print the console line.
            System.out.print("00000119> ");
            
            //Read the input from user.
            commandLine = console.readLine();
            
            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if(commandSplit.length > 1)
            {
                //Add the parameters into commandParams
                for(int i = 1; i < commandSplit.length; i++)
                {
                    commandParams.add(commandSplit[i]);
                }
            }
            
            switch(commandSplit[0])
            {
                case "about" :
                               //Output if user enters "about" command.
                               commandOutput.add(commandLine);
                               System.out.println("Author: " +name);
                               System.out.println("Version: " +version);
                               break;           
                case "time" :
                                //Output if user enters "time" command.
                                commandOutput.add(commandLine);
                                if (commandSplit[0].equals("time")) {
                                Date time = new Date();
                               //Format the output in hour:minute:second AM/PM format
                                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                                System.out.println(sdf.format(time));
                                 }
                                break;
                case "date" :
                                //Output if user enters "date" command.
                                commandOutput.add(commandLine);
                                if (commandSplit[0].equals("date")) {
                                Date date = new Date();
                               //Format the output in day/month/year format
                                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                                System.out.println(sdf.format(date));}
                                break;    
                 case "ls" :
                                    //To retrieve files and folders (or directories) from the current directory
                                    if (commandSplit[0].equals("ls")) {
                                        
                                        String current = new java.io.File( "." ).getCanonicalPath();
                                        System.out.println("Current dir:"+current);  
                                    }
                                    
                                    break;          
                case "exit"  : 
                               //Exit the console when "exit" is entered.
                               System.exit(0);
                               break;                                
                case ""      : 
                               //Reprint the console line if user just pressed "enter" without any command.
                               break;            
                case "hist"  : 
                               //Example of a built-in function that can be used to trace the command history. 
                               if (commandParams.size() < 1)
                               {
                                  System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                                  break;
                               }                                                 
                               //Process based on the number of parameters attached to the command.
                               if(commandParams.get(0).equals("-h"))
                               {
                                   commandOutput.add(commandLine);
                                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                                            System.out.println("hist -c : Clear the command history.");
                                            break;
                               }
                               
                               if(commandParams.get(0).equals("-c") && commandLine.length() < 10)
                               {
                                    if(commandOutput.isEmpty())
                                        {
                                            System.out.println("No record to clear from empty history.");
                                        }       
                                    
                                   commandOutput.add(commandLine);
                                   commandOutput.clear();
                                   
                                                              
                                   break;
                               }
                               else
                                    {
                                        if(commandParams.get(0).equals("-c") && commandParams.get(1).equals("-l"))
                                            {
                                            number  = Integer.parseInt((String) commandParams.get(2));

                                            if(number <= 0)
                                                {
                                                    System.out.println("Invalid Number.");
                                                    break;
                                                }

                                            commandOutput.add(commandLine);
                                            commandOutput.clear();
                                            System.out.println("History has been deleted.");
                                            break;
                                            }
                                    }
                               
                               if(commandParams.get(0).equals("-l") && commandLine.length() < 11)
                               {
                                   number = Integer.parseInt((String) commandParams.get(1));
                                   
                                   if(number < 0 )
                                        {
                                            System.out.println("Invalid Number.");
                                        }
                                        else
                                            {
                                                if(commandOutput.isEmpty())
                                                    {
                                                        commandOutput.add(commandLine);
                                                        //System.out.println("No command previously recorded in history.");
                                                        continue;
                                                    }
                                       
                                                    if(commandOutput.size() >= number)
                                                         {
                                                             for(int x = commandOutput.size() - number; x < commandOutput.size(); x++)
                                                             {
                                                                 System.out.println(commandOutput.get(x));
                                                             }
                                                         }
                                                    else
                                                         {
                                                             for(int x = 0; x < commandOutput.size(); x++)
                                                                 {
                                                                     System.out.println(commandOutput.get(x));
                                                                 }
                                                         }
                                             }
                                   
                                   commandOutput.add(commandLine);
                                   break;
                               }
                               else
                               {
                                   if(commandParams.get(0).equals("-l") && commandParams.get(2).equals("-c"))
                                   {
                                       number = Integer.parseInt((String) commandParams.get(1));
                                       
                                       if(number < 0 )
                                            {
                                                System.out.println("Invalid Number.");
                                            }
                                       else
                                       {
                                           if(commandOutput.isEmpty())
                                           {
                                               commandOutput.add(commandLine);
                                               commandOutput.clear();
                                               System.out.println("No command recorded in history.");
                                               continue;
                                           }
                                           
                                           if(commandOutput.size() >= number)
                                           {
                                               for(int x = commandOutput.size() - number; x < commandOutput.size(); x++)
                                               {
                                                   System.out.println(commandOutput.get(x));
                                               }
                                           }
                                           else
                                           {
                                               for(int x = 0; x < commandOutput.size(); x++)
                                               {
                                                   System.out.println(commandOutput.get(x));
                                               }
                                           }
                                       }

                                       commandOutput.add(commandLine);
                                       commandOutput.clear();
                                       break;
                                   }
                               }
                               
                               //Checks if condition is TRUE.
                               if(commandParams.get(0) != "-h" || commandParams.get(0) != "-c" || commandParams.get(0) != "-l")
                                    {
                                        System.out.println("Invalid Command.");
                                        break;
                                    }
                case "NOTEPAD" :              
                case "notepad" :
                case "Notepad" :
                               commandOutput.add(commandLine);
                               ProcessBuilder Notepad = new ProcessBuilder("notepad.exe");
                               Notepad.start();
                               break;
                    
                default      : 
                               System.out.println("Invalid command.");
                               break;
            }
        }
    }
}
