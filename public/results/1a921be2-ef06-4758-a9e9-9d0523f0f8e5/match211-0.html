package myshell;
import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000117 {

	public static void main(String[] args) throws IOException {
		
        String commandLine;
        ArrayList commandOutput = new ArrayList(); 
        
        int number = 0;
        String name = "Kevin";
        double version = 0.01;
        
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("My Shell Version 0.01");
        
        while (true) {
            System.out.print("00000117> ");
            commandLine = console.readLine();
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            if (commandSplit.length > 1) {
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }
            
            if (commandSplit[0].equals("")) {
                continue;
            }
            if (commandSplit[0].equals("time")) {
                Date time = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
            }
                        
            if (commandSplit[0].equals("date")) {
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
            }
            
            if (commandSplit[0].equals("exit")) {
                System.exit(0);
                break;
            }
            
            if (commandSplit[0].equals("about")) {
            	System.out.println("Author: " +name);
                System.out.println("Version: " +version);               
            }
            
            if (commandSplit[0].equals("ls")) {
            	System.out.println("test_folder"); 
            	System.out.println("test_file.txt");                
            }
            
            if (commandSplit[0].equals("notepad")) {
            	ProcessBuilder Notepad = new ProcessBuilder("notepad.exe"); 
                Notepad.start(); 
                break;
            }
            
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    break;
                } 
                if(commandOutput.isEmpty())
                {
                    commandOutput.add(commandLine);
                    System.out.println("No command previously recorded in history.");
                    continue;
                } 
                if(commandParams.get(0).equals("-h"))
        {
            commandOutput.add(commandLine);
            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
            System.out.println("hist -c : Clear the command history.");
            break;
        }
        
        if(commandParams.get(0).equals("-c") && commandLine.length() < 10)
        {
            
            
            if(commandOutput.isEmpty())
            {
                System.out.println("No record to clear from empty history.");
            }
             
            commandOutput.add(commandLine);
            commandOutput.clear(); 
            
            break;
        }
        else
        {
            if(commandParams.get(0).equals("-c") && commandParams.get(1).equals("-l"))
            {
            number  = Integer.parseInt((String) commandParams.get(2));
            
            if(number <= 0)
            {
                System.out.println("Invalid Number.");
                break;
            }
            
            commandOutput.add(commandLine);
            commandOutput.clear();
            System.out.println("History has been deleted.");
            break;
            }
        }
        
        if(commandParams.get(0).equals("-l") && commandLine.length() < 11)
        {
            number = Integer.parseInt((String) commandParams.get(1));
            
            if(number < 0 )
            {
                System.out.println("Invalid Number.");
            }
            else
            {
                if(commandOutput.isEmpty())
                {
                    commandOutput.add(commandLine);
                    System.out.println("No command previously recorded in history.");
                    continue;
                }
                
                if(commandOutput.size() >= number)
                {
                    for(int x = commandOutput.size() - number; x < commandOutput.size(); x++)
                    {
                        System.out.println(commandOutput.get(x));
                    }
                }
                else
                {
                    for(int x = 0; x < commandOutput.size(); x++)
                    {
                        System.out.println(commandOutput.get(x));
                    }
                }
            }
            
            commandOutput.add(commandLine);
            break;
        }
        else
        {
            if(commandParams.get(0).equals("-l") && commandParams.get(2).equals("-c"))
            {
                number = Integer.parseInt((String) commandParams.get(1));
                
                if(number < 0 )
                {
                    System.out.println("Invalid Number.");
                }
                else
                {
                    if(commandOutput.isEmpty())
                    {
                        commandOutput.add(commandLine);
                        commandOutput.clear();
                        System.out.println("No command previously recorded in history.");
                        continue;
                    }
                    
                    if(commandOutput.size() >= number)
                    {
                        for(int x = commandOutput.size() - number; x < commandOutput.size(); x++)
                        {
                            System.out.println(commandOutput.get(x));
                        }
                    }
                    else
                    {
                        for(int x = 0; x < commandOutput.size(); x++)
                        {
                            System.out.println(commandOutput.get(x));
                        }
                    }
                } 
                 
                         
          if(commandParams.get(0) != "-h" && commandParams.get(0) != "-c" && commandParams.get(0) != "-l")
         {
            System.out.println("Invalid Command.");
            break;
         }
                                       
                commandOutput.add(commandLine);
                commandOutput.clear();
                break;  
                
                }
            
            
            }
        
        } 
            
      }
    }
 }
