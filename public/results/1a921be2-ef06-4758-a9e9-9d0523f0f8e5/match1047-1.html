import java.io.File;
import java.text.DateFormat;

import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Date;
import java.util.Scanner;

public class myshell_00000046 {
    static Scanner in = new Scanner(System.in);
    static ArrayList<String> history = new ArrayList<>();
    
    public static void main(String[] args) {
        String command;
        
        println("Welcome to myShell by NG ZHEN HOU, v0.01");
        do {
            command = mainMenu();
        } while(!command.equals("exit"));
        
        in.close();
        println("Exiting shell by NG ZHEN HOU, thanks for using the shell!");
    }

    private static String mainMenu() {
        printf("00000046 > ");
        String[] command = in.nextLine().toLowerCase().split(" ");
        
        // Straight away exit if exiting
        if(command[0].equals("exit"))
            return command[0];
        
        switch(command[0]) {
            case "about":
                printAbout();
                history.add(String.join(" ", command));
                break;
            case "hist":
                try {
                    historyFunction(command);
                    history.add(String.join(" ", command));
                } catch(IllegalArgumentException | ArrayIndexOutOfBoundsException ex) {
                    printf("Invalid argument for hist,\n"
                            + "-c   parameter to clear history\n"
                            + "-l x parameter to list last x history\n");
                }
                break;
            case "ls":
                lsFunction(command);
                history.add(String.join(" ", command));
                break;
            case "notepad":
                openNotepad();
                history.add(String.join(" ", command));
                break;
            case "time":
                printTime();
                history.add(String.join(" ", command));
                break;
            case "date":
                printDate();
                history.add(String.join(" ", command));
                break;
            case "exit":
                break;
            default:
                println("Invalid command, available commands:\n"
                        + "about, date, hist, ls, notepad, time, exit");
        }
        
        println("-----");
        return command[0];
    }
    
    private static void printDate() {
        DateFormat formatDate = new SimpleDateFormat("dd MMM yyyy");
        println("Date today: " + formatDate.format(new Date()));
    }

    private static void printTime() {
        DateFormat formatTime = new SimpleDateFormat("hh:mm:ss a");
        println("Time now: " + formatTime.format(new Date()));
    }

    private static void openNotepad() {
        ProcessBuilder notepadProcess = new ProcessBuilder("Notepad.exe");
        
try {
            notepadProcess.start();
        }
        catch(Exception ex) {
            ex.printStackTrace();
        }
    }

    private static void lsFunction(String[] commandArgs) {
        String[] validFlags = {"d", "f"};
        
        File currentDirectory = new File(".");
        File[] listOfFiles = currentDirectory.listFiles();
        
        if(commandArgs.length == 1) {
            for(File file : listOfFiles) {
                if(file.isDirectory()) {
                    println("D " + file.getName());
                }
                else {
                    println("  " + file.getName());
                }
            }
        }
        else if(commandArgs.length > 2) { // If there are more than one flag
            println("ls command only accepts one flag at a time:\n"
                    + "-d for directory list only\n"
                    + "-f for file list only");
        }
        else if(Arrays.stream(validFlags).anyMatch(commandArgs[1].replaceAll("-", "")::equals)) 
{
            if(commandArgs[1].replaceAll("-", "").equals("d")) {
                for(File file : listOfFiles) {
                    if(file.isDirectory()) {
                        println("D " + file.getName());
                    }
                }
            }
            else 
if(commandArgs[1].replaceAll("-", "").equals("f")) {
                for(File file : listOfFiles) {
                    if(file.isFile()) {
                        println(file.getName());
                    }
                }
            }
        }
        else {
            println("Invalid flag, flags available:\n"
                    + "-d for directory list only\n"
                    + "-f for file list only");
        }
    }

    private static void historyFunction(String[] commandArgs) {
        String[] validFlags = {"l", "c"};
        
        if(findDuplicate(commandArgs)) {
            throw new IllegalArgumentException();
        }
        
        if(commandArgs.length == 1) {
            println("Printing all previous commands used (oldest to most recent):");
            for(String command : history) {
                println(command);
            }
        }
        else if(Arrays.stream(validFlags).anyMatch(commandArgs[1].replaceAll("-", "")::equals)){
            for(int i = 1; i < commandArgs.length; i++) {
                if(commandArgs[i].replaceAll("-", "").equals("c")) {
                    println("Clearing command history.");
                    history.clear();
                }
                else if(commandArgs[i].replaceAll("-", "").equals("l")) {
                    if(Integer.parseInt(commandArgs[i + 1]) > 0) {
                        int list = Integer.parseInt(commandArgs[i + 1]);
                        if(list < 1) {
                            println("-l parameter must be more than 0.");
                        }
                        else if(history.size() < list) {
                            println("History is too small from the -l parameter, try a smaller number.");
                        }
                        else {
                            for(int j = history.size() - list; j < history.size(); j++) {
                                println(history.get(j));
                            }
                        }
                    }
                    else {
                        throw new IllegalArgumentException();
                    }
                }
            }
        }
        else {
            throw new IllegalArgumentException();
        }
    }

    private static void printAbout() {
        println("Shell by NG ZHEN HOU");
        println("Version: 0.01");
    }

    // CONVENIENCE METHODS, IGNORE THEM
    private static void println(Object arg) {
        System.out.println(arg);
    }
    
    private static void printf(String string, Object... args) {
        System.out.printf(string, args);
    }
    
    private static boolean findDuplicate(String[] arg) {
        for(int i = 0; i < arg.length - 1; i++) {
            for(int j = i + 1; j < arg.length; j++) {
                if(arg[i].equals(arg[j])) {
                    return true;
                }
            }
        }
        
        return false;
    }
}
