package myshell;

import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000004 
{
	public static void main(String[] args) throws IOException 
	{
    	int flagflag = 0;
        String commandLine; //Use commandLine to store the command and the argument(s)
        
BufferedReader console = new BufferedReader(new InputStreamReader(System.in));  //Read and buffered the command and the argument(s)
        System.out.println("My Shell Version 0.01");  //Just a banner
        ArrayList<String> commandHist =  new ArrayList<String>();
        
        while (true) 
        {
        	flagflag = 0;  //Print the console line
            System.out.print("00000004> "); //Remember to replace "student_id" with your student number
            commandLine = console.readLine(); //Read the input from user

            //Split the input into command and parameters.
            ArrayList commandParams = new ArrayList();  //commandParams is a list that is pre-created to store all the parameters
            String[] commandSplit = commandLine.split(" ");  //commandSplit is a string array that is used to store the command and the parameters after splitting them
            
            //Check to see if there is any parameter.
            if (commandSplit.length > 1) //If size of commandSplit is larger than one that means the user has included some parameters in the input
            { 
                for (int i = 1; i < commandSplit.length; i++) //Add the parameters into commandParams
                {
                    commandParams.add(commandSplit[i]);
                }
            }
 
            if (commandSplit[0].equals("")) //Reprint the console line if user just pressed "enter" without any command
            {
                continue;
            }

            if (commandSplit[0].equals("time")) //Print the current time
            {
                Date time = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a"); //Format the output in hour:minute:second AM/PM format
                System.out.println(sdf.format(time));
                commandHist.add(commandLine);
                flagflag = 1;
            }
            
            if (commandSplit[0].equals("date")) //Print today's date
            {
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy"); //Format the output in day/month/year format
                System.out.println(sdf.format(date));
                commandHist.add(commandLine);
                flagflag = 1;
            }
            
            if (commandSplit[0].equals("exit")) //Exit the console when "exit" is entered
            {
                System.exit(0);
                flagflag = 1;
            }
            
            if (commandSplit[0].equals("ls")) //To retrieve files and folders (or directories) from the current directory
            { 	
            	if(commandParams.size() < 1) 
            	{
            		File folder = new File(".");
                    File[] listOfFiles = folder.listFiles();
                    flagflag = 1;
                    for (File a : listOfFiles) 
                    {
                    	System.out.println(a.getName());
                    }
                    commandHist.add(commandLine);
            	}         
            	else 
            	{
	                if(commandParams.get(0).equals("-d")) 
	                {
	                	File folder = new File(".");
	                    File[] listOfFiles = folder.listFiles();
	                    flagflag = 1; 
	                    for (Fi
le a : listOfFiles) 
	                    {
	                        if (a.isDirectory()) 
	                        {
	                           System.out.println(a.getName());  
	                        }
	                    }
	                    commandHist.add(commandLine);
	                }
	                else if(commandParams.get(0).equals("-f"))
	                {
	                	File folder = new File(".");
	                    File[] listOfFiles = folder.listFiles();
	                    flagflag = 1;	              
	                    for (Fi
le a : listOfFiles) 
	                    {    
	                        if (a.isFile()) 
	                        {
	                           System.out.println(a.getName());  
	                        }    
	                    }
	                    commandHist.add(commandLine);
	                }	                              
            	}
            }
            
            if (commandSplit[0].equals("hist")) //Example of a built-in function that can be used to print the command history
            {          		
                if (commandParams.size() < 1) 
                {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    flagflag = 1;
                    continue;
                } 
                else 
                {       
                    if (commandParams.get(0).equals("-h")) //Process based on the number of parameters attached to the command
                    {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                        flagflag = 1;
                        commandHist.add(commandLine);
                        continue;
                    }
                    
	                if (commandParams.get(0).equals("-c")) 
	                {	                    	
	                	flagflag = 1;
	                    System.out.println("Deleted all command history.");
	                    commandHist.clear();	
	                }

                }                   
            }
            
            if (commandSplit[0].equals("about")) //Display the author & the version number
            {
            	System.out.println("Author: Syahmi Shamsul");
            	System.out.println("Version: 0.01");
            	commandHist.add(commandLine);
            	flagflag = 1;
            }
            
            if (commandSplit[0].equals("notepad")) //Process to launch Notepad
            {
            	ProcessBuilder pobuild = new ProcessBuilder("Notepad.exe");
            	pobuild.start();
            	commandHist.add(commandLine);
            	flagflag = 1;
            
}
                      
            if ( flagflag == 0) //User enters a command that is not in the list
            {
            	System.out.println("Invalid Command! - This is not a valid command.");
            }          
        }        
    }
}
