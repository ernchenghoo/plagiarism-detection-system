public class myshell_00000142 {
    
    public static void main(String[] args) throws IOException {
        //history is a list that stores all the command
        ArrayList history = new ArrayList();        
        
        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("My Shell Version 0.01");

        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000142> ");

            //Read the input from user.
            commandLine = console.readLine();

            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            ArrayList commandParams = new ArrayList();
            
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.           
            String[] commandSplit = commandLine.split(" ");
                                    
            // current directory
            File curentfile = new File(".");                                 
            
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);                 
                }
            }
            
            //empty input does not add into history
            if (!"".equals(commandLine)){
                history.add(commandSplit);   
            }

            //use switch case to select input
            String input = commandSplit[0];            
            switch(input){
                
                //Reprint the conline line if user just pressed "enter" without any command.
                case "":
                    break;
                
                //Print the current time.
                case "time":                    
                    Date time = new Date();
                    //Format the output in hour:minute:second AM/PM format
                    SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                    System.out.println(sdf.format(time));
                    break;
                    
                //Print today's date.    
                case "date":
                    Date date = new Date();
                    //Format the output in day/month/year format
                    sdf = new SimpleDateFormat("E dd/MM/yyyy");
                    System.out.println(sdf.format(date));
                    break;
                    
                //Print about
                case "about":
                    System.out.println("author: Chan Wai Loon");
                    System.out.println("version: 0.01");    
                    break;
                    
                //Exit the console when "exit" is entered.
                case "exit":
                    System.exit(0);
                    break;
                
                //launch notepad.     
                case "notepad":
                    Runtime rs = Runtime.getRuntime();
                    rs.exec("notepad");
                    break;
                    
                //To retrieve files and folders (or directories) from the current directory
                case "ls":
                    File[] files = curentfile.listFiles();
                    try{
                        //print only files
                        if (commandParams.get(0).equals("-f")) {
                            for(File f : files){
                                if(f.isFile()){
                                    System.out.println(f.getName());
                                }
                            }
                        }
                        //print only directories
                        if (commandParams.get(0).equals("-d")) {
                            for(File f : files){
                                if(f.isDirectory()){
                                    System.out.println(f.getName());
                                }
                            }
                        }
                    }
                    catch(Exception e){
                        //print both files and directories if no argument is entered
                        for(File f : files){
                            if(f.isFile()){
                                System.out.println(f.getName());
                            }
                            if(f.isDirectory()){
                                System.out.println(f.getName());
                            }
                        }
                    }
                    break;
                
                //To print the command history.     
                case "hist":
                    //If size of commandParam is lesser than one that means the user has not included any parameter in the input.                
                    if (commandParams.size() < 1) {
                        //Show error
                        System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    }
                    else {
                        //Process based on the number of parameters attached to the command.
                        if (commandParams.get(0).equals("-h")) {
                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                            System.out.println("hist -c : Clear the command history.");
                        }
                        //If size of commandParam is lesser than 2 that means the user has only included parameter but no integer in the input.
                        else if (commandParams.get(0).equals("-l")) {    
                            try{
                                //read the integer
                                int historyInteger = Integer.valueOf((String)commandParams.get(1)) ;  
                                int historySize = history.size();
                                try{
                                    for (int i = historySize -(historyInteger); i < history.size(); i++) {                            
                                        //print output without brackets and commas
                                        System.out.println(Arrays.toString((Object[])history.get(i-1)).replace("[","").replace("]","").replace(",",""));
                                    }
                                    
                                    if(commandParams.get(2).equals("-c")) {
                                        System.out.println("Deleted: All command history.");
                                        //clear history array list
                                        history.clear();
                                    }
                                }
                                //If there is no previous history or history out of bound
                                catch (ArrayIndexOutOfBoundsException aiofb){
                                    int entryDisplay = historySize-1;
                                    if (entryDisplay == 0){
                                        System.out.println("There is no entry");
                                    }
                                    else if (entryDisplay == 1){
                                        System.out.println("There is only "+ entryDisplay +" entry");
                                    }
                                    else if (entryDisplay >1 ){
                                        System.out.println("There are only "+ entryDisplay + " entries");
                                    }                                
                                }  
                                //Show error if there is no integer
                                if (commandParams.size() < 2) {
                                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                                }
                                else{
                                    try{
                                        //If integer is negative value show an error
                                        if (Integer.valueOf((String)commandParams.get(1))<=0 ){
                                            System.out.println("Invalid number");                            
                                        }
                                    }
                                    //If input is not an integer show an error
                                    catch (NumberFormatException nfe){
                                        System.out.println("Invalid number");
                                    }  
                                }    
                            }
                            //If x is not an integer show an error
                            catch (Exception nfe){
                                System.out.println("invalid input");
                            }                                                                                                  
                        }                   
                        else if (commandParams.get(0).equals("-c")) {
                            System.out.println("Deleted: All command history.");
                            //clear history array list
                            history.clear();
                        }
                        else {
                            //print error if argument is not in the list
                            System.out.println("Invalid input argument.");
                        }
                    }
                    break;
                    
                default:
                    //print error when input is not in the list
                    System.out.println("Invalid command!");
            }           
        }
    }
}
