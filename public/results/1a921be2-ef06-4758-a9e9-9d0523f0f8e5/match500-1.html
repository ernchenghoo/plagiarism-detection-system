import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;
import java.util.Scanner;

public class myshell_00000054 {
    
    public static void main(String[] args) throws java.io.IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;
        ArrayList<String> commandStore = new ArrayList<>();
        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        Scanner osDecide = new Scanner(System.in);
        //Just a banner
        System.out.println("***********************");
        System.out.println("My Shell Version 0.01");
        System.out.println("***********************");
        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell
        while (true) {
            //Print the console line.
            System.out.print("00000054> ");
            //Read the input from user.
            commandLine = console.readLine();
            //Split the input into command and parameters.
            //commandStore is a list that saves every command entered by the user 
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList<>();
            String[] commandSplit = commandLine.split(" ");
            File folder = new File(".");
            File[] listOfFiles = folder.listFiles();
            //Save the user entered command line to the array
            if (!(commandSplit[0].equals("hist"))){
                commandStore.add(commandLine);
            }
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }               
            }
            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }
            //Print the author information
            else if (commandSplit[0].equals("about")){
                System.out.println("Author: Walid Mohamed");
                System.out.println("Version Number: 0.01");
            }
            //Print the current time.
            else if (commandSplit[0].equals("time")) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
            }            
            //Print today's date.
            else if (commandSplit[0].equals("date")) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
            }            
            //Exit the console when "exit" is entered.
            else if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }          
            //To retrieve files and folders (or directories) from the current directory
            else if (commandSplit[0].equals("ls")) {
                if (commandParams.size()==0){
                    for (int i = 0; i < listOfFiles.length; i++) {
                        if (listOfFiles[i].isFile()) {
                            System.out.println("File " + listOfFiles[i].getName());
                            for (int b = 0; b < listOfFiles.length; b++) 
{
                                if (listOfFiles[b].isDirectory()) {
                                    System.out.println("Directory " + listOfFiles[b].getName());
                                }                 
                            } 
                        }
                    }                    
                }
                else if (commandParams.get(0).equals("-f")) {
                    
for (int i = 0; i < listOfFiles.length; i++) {
                        if (listOfFiles[i].isFile()) {
                            System.out.println("File " + listOfFiles[i].getName());
                        }
                    }
                }
                else if(commandParams.get(0).equals("-d")){
                    
for (int b = 0; b < listOfFiles.length; b++) {
                        if (listOfFiles[b].isDirectory()) {
                            System.out.println("Directory " + listOfFiles[b].getName());
                    }                         
                }   
                }else{
                    for (int i = 0; i < listOfFiles.length; i++) {
                        if (listOfFiles[i].isFile()) {
                            System.out.println("File " + listOfFiles[i].getName());
                            for (int b = 0; b < listOfFiles.length; b++) {
                                if (listOfFiles[b].isDirectory()) {
                                    System.out.println("Directory " + listOfFiles[b].getName());
                                }                 
                            } 
                        }
                    }
                }
           
            }            
            //Example of a built-in function that can be used to print the command history. 
            else if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } 
                else {
                    //Process based on the number of parameters attached to the command.
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                    }
                    if (commandParams.get(0).equals("-l")){
                        if (commandParams.size() < 2){
                            System.out.println("No valid parameters entered");
                        }
                        else if(commandParams.size() >= 1) {
                            int number = Integer.parseInt(commandParams.get(1).toString());
                            if (number > 0){
                            //loop the number the user input in the command.
                            //print top-down, the last [number] of commands in history.
                                for (Integer h = 0; h < number; h++) {
                                    //if the index is invalid continue, to avoid out of index errors.
                                    //incase the use input "-l 50" and there not at least 50 commands in the history 
                                    if (number > commandStore.size()){
                                        System.out.println("Command history is less than the input value");
                                        break;
                                    }
                                    else{
                                        System.out.println(commandStore.get(h));
                                    }
                                }
                            
}
                        
                        //if the number is not positive output "Invalid Number".
                        }else{
                            System.out.println("Invalid Number");
                    }
                }                   
                //if the parameter "-c" is found any where clear the command history.
                if (commandParams.get(0).equals("-c")) {
                    System.out.println("Deleted: All command history.");
                    commandStore.clear(); //clear "commandsHist" array which contains the history
                    System.out.println("Command history has been deleted.");
                
}
        }
           }               
            //Start the notepad process on windows
            else if (commandSplit[0].equals("notepad")){
                System.out.print("Are you using Windows OS? Y/n\n");
                String decision = osDecide.next();
                //If OS used is not windows, do not run the process to avoid crashing
                
if(decision.equals("n")){
                    System.out.println("Notepad is not availabe, use gedit for Linux");
                    
                }else if(decision.equals("Y")){
                    ProcessBuilder proc = new ProcessBuilder("notepad.exe"); //"notepad.exe" is the name of the process.
                    proc.start(); //start the process.
                    break; 
                }else{
                    System.out.println("Invalid Command");
                }
            }
            else if (commandSplit[0].equals("gedit")){
                ProcessBuilder proc = new ProcessBuilder("gedit"); //"gedit" is the name of the process in ubuntu.
                proc.start(); //start the process.
                break;
            
}
            else {
                System.out.println("Invalid Command");
            }
        }
    }
}
