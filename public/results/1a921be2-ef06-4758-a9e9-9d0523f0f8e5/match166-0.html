import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000161 {
    
    public static void main(String[] args) throws IOException {

        String commandLine;

        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("My Shell Version 0.01");
        
         ArrayList clearHistory = new ArrayList ();
        

        while (true) {
            
            System.out.print("00000161> ");

            commandLine = console.readLine();

            ArrayList commandParams = new ArrayList();
           
            String[] commandSplit = commandLine.split(" ");
            ArrayList<String> acceptedComms = new ArrayList<>();
            int count = 0;

            if (commandSplit.length > 1) {
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            if (commandSplit[0].equals("")) {
                continue;
            }

            if (commandSplit[0].equals("time")) {
                Date time = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                count++;
                
            }
            
            if (commandSplit[0].equals("date")) {
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                count++;
            }
            //about
            if (commandSplit[0].equals("about")){
                System.out.println("author: Derrick Chia Thin Choon");
                System.out.println("version: 0.01");
                count++;
            }
            
            //notepad
            if (commandSplit[0].equals("notepad")){
                Runtime rs = Runtime.getRuntime();
 
                try {
                rs.exec("notepad");
                }
                catch (IOException e) {
                System.out.println(e);
                }
                count++;
            }
            if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }
            //list
            if (commandSplit[0].equals("ls")) {
                if(commandParams.size() <1){
                   count++;
                   File files = new File (".");
                    File[] fileList = files.listFiles();
                     for(File file : fileList) {
                        if(file.isFile()|| file.isDirectory()){
                            System.out.println(file.getName());
                }
            }
            }
                else{
                    if(commandParams.get(0).equals("-f")){
                        count++;
                        File files = new File (".");
                            File[] fileList = files.listFiles();
                                for(File file : fileList) {
                                    if(file.isFile()){
                                        System.out.println(file.getName());
                }
            }
                    }
                    if(commandParams.get(0).equals("-d")){
                        count++;
                        File files = new File (".");
                            File[] fileList = files.listFiles();
                                for(File file : fileList) {
                                    if(file.isDirectory()){
                                        System.out.println(file.getName());
                }
            }
                    }
                }
            }
            
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    count=count+2;
                } else {
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                        count++;
                    }
                    if(commandParams.size() == 1){
                    if (commandParams.get(0).equals("-c")) {
                        System.out.println("Deleted: All command history.");
                        clearHistory.clear();
                        count=count+2;
                    }
                    }
                    if (commandParams.size()>1){
                            if (commandParams.get(0).equals("-c") && commandParams.get(1).equals("-l")){
                                clearHistory.clear();
                                count=count+2;
                                try{
                                   int counter = Integer.parseInt(commandParams.get(2).toString());
                                       for (int j =0; j<counter; j++){
                                           if (j >= clearHistory.size()){
                                               break;
                                           }
                                           System.out.println(clearHistory.get(j));
                                       }
                               }catch(NumberFormatException e){
                                   System.out.println("You can't do that");  
                                   count=count+2;
                               }                               
                            }
                        }
                       
                        if (commandParams.get(0).equals("-l")){
                            if (commandParams.size() > 1){
                            count++;
                            try{
                                int counter = Integer.parseInt(commandParams.get(1).toString());
                                for (int k =0; k<counter; k++){
                                    if (k >= clearHistory.size()){
                                        break;
                                    }
                                System.out.println(clearHistory.get(k));
                            }
                            }catch(NumberFormatException e){
                                System.out.println("You can't do that");
                                count++;
                            }
                            if (commandParams.size() > 2){
                            if (commandParams.get(2).equals("-c")){
                                clearHistory.clear();
                                count=count+2;
                            }
                            else {
                                System.out.println("You can't do that"); 
                                count=count+2;
                            }
                            }
                            }
                        }
                     
            }
            
                
            
        }
            acceptedComms.add("time");
            acceptedComms.add("date");
            acceptedComms.add("about");
            acceptedComms.add("notepad");
            acceptedComms.add("ls");
            acceptedComms.add("hist");
            acceptedComms.add("exit");
            
           boolean accepted = false;
            for(int l = 0; l < acceptedComms.size(); l++) {
                if(commandSplit[0].equals(acceptedComms.get(l))) {
                    accepted = true;
                }
            }
            
            if(accepted == false) {
                System.out.println("Invalid command");
            }
                      clearHistory.add(commandLine);  

            }
    }
   
    }
