package myshell;

import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000170 {
    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;
        boolean status = false;
        
        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        ArrayList historyCommand = new ArrayList();
        //Just a banner
        System.out.println("My Shell Version 0.01");
        
        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) 
        {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000170 > ");

            //Read the input from user.
            commandLine = console.readLine();
          
            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");

            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            //If has space than jump here
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {                    
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                System.out.println("Please enter any command");
                status = false;
                continue;
            }

            //Print the current time.
            if (commandSplit[0].equals("time")) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                commandParams.add(commandLine);
                status = true;
            }

            //Print today's date.
            if (commandSplit[0].equals("date")) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                commandParams.add(commandLine);
                status = true;
            }

            //Exit the console when "exit" is entered.
            if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }

            //To retrieve files and folders (or directories) from the current directory
            if (commandSplit[0].equals("ls")) {                
                if(commandParams.size() < 1) {
                    System.out.println("The path of the directories and file is list at below");
                    
                    File currentDirFile = new File(".");
                    File[] listOfFiles = currentDirFile.listFiles();
                    String helper = currentDirFile.getAbsolutePath();
                    
                    //List the file and directories                    
                    System.out.println(helper);
                        String[] files = currentDirFile.list();
                        if (files.length == 0) {
                            System.out.println("The directory is empty");
                        }
                        else
                        {
                            int count = 1;
                            for(String aFile : files){
                                System.out.println(count + ") " + aFile);
                                count++;
                            }
                        }                                     
                    status = true;
                }
                else {
                    if(commandParams.get(0).equals("-f")) {
                        System.out.println("The path of the file is list at below");
                        File currentDirFile = new File(".");
                        File[] listOfFiles = currentDirFile.listFiles();
                        int count = 1;
                        for(File file: listOfFiles) {
                            if(file.isFile()) {
                                System.out.println(count + ") " + file.getName());
                                count++;
                            }
                        }
                        status = true;
                    }
                    if(commandParams.get(0).equals("-d")) {
                        System.out.println("The path of the directories is list at below");
                        File currentDirFile = new File(".");
                        String[] files = currentDirFile.list();
                        if (files.length == 0) {
                            System.out.println("The directory is empty");
                        }
                        else
                        {
                            File[] listOfFiles = currentDirFile.listFiles();  
                            int count = 1;
                            for(int i = 0; i < listOfFiles.length; i++){
                                if(listOfFiles[i].isDirectory()){
                                    System.out.println(count + ") " + listOfFiles[i].getName());
                                    count++;
                                }
                            }
                        }                    
                        status = true;
                    }
                }
            }

            //Example of a built-in function that can be used to print the command history. 
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    status = true;
                } 
                else {
                    //Process based on the number of parameters attached to the command.
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                        status = true;
                    }
                    if (commandParams.get(0).equals("-c")) {
                        System.out.println("Deleted: All command history.");
                        historyCommand.clear();
                        status = false;
                    }                   
                    if (commandParams.get(0).equals("-l")) { 
                        if(commandParams.size() < 2){
                            System.out.println("Invalid Command!");
                            status = true;
                        }
                        else {                     
                            int numhistory; 
                            boolean check = false;
                            try {
                                numhistory = Integer.parseInt(commandParams.get(1).toString());
                                if(numhistory==0 || numhistory<0) {
                                    System.out.println("Invalid Number");
                                    status = true;
                                    check = true;
                                }
                                if(numhistory>historyCommand.size()) {
                                    System.out.println("Your request the number of command history is excess.");
                                    status = true;
                                    check = true;
                                }                                    
                            }
                            catch(NumberFormatException ex) {
                                
                                System.out.println("Invalid command!");
                                historyCommand.add(commandLine);
                                continue;
                            }
                            if(check != true) {
                                System.out.println("This is your " + numhistory + " previously command[s].");
                                int count = 1;
                                for(int i = historyCommand.size(); i > historyCommand.size()-numhistory; i--){                                                                  
                                    System.out.println(count+ ") " +historyCommand.get(i-1));
                                    count++;                                   
                                }
                                status = true;
                            }                                
                        }     
                    }
                }               
            }
            
            if (commandSplit[0].equals("about")) {
                System.out.println("Author: Lam Yi Feng");
                System.out.println("Version: 0.01");
                commandParams.add(commandLine);
                status = true;
            }
            //Execute the notepad
            if (commandSplit[0].equals("notepad")) {
                Runtime pb = Runtime.getRuntime();
                try {
                    pb.exec("notepad");
                    status = true;
                }
                catch(IOException e) {
                    System.out.println("This is invalid command");
                    status = true;
                }
            }
            //Check the validation Input
            if(!commandSplit[0].equals("date")&&!commandSplit[0].equals("time")&&!commandSplit[0].equals("about")&&!commandSplit[0].equals("ls")&&!commandSplit[0].equals("notepad")&&!commandSplit[0].equals("hist")) {
                System.out.println("Invalid Command!");
                status = true;
            }
            //Store the command line to the historyCommand array
            if(status == true) {
                historyCommand.add(commandLine);
            }
        }
    }
}
