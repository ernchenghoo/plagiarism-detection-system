import java.io.*;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Date;

/**
 *
 * @author STAN-PC
 */
public class myshell_00000067 {
    
    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;
        
        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        
        ArrayList command = new ArrayList(); //To store user command input
        
        //Just a banner
        System.out.println("My Shell Version 0.01");
        
        while (true) {
            //Print the console line.
            System.out.print("00000067 > ");
            
            //Read the input from user.
            commandLine = console.readLine();
            
            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = 
commandLine.split(" ");
            ArrayList<String> cmd = new ArrayList<>(); //To store valid command inputs 
            
            //Add valid command inputs into array
            cmd.add("about");
            cmd.add("time");
            cmd.add("date");
            cmd.add("hist");
            cmd.add("ls");
            cmd.add("notepad");
            cmd.add("exit");
            
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }
            
            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }
            
            //Print about details 
            if (commandSplit[0].equals("about")) {
                command.add(commandLine); //Add input into array
                System.out.println("author: Stanley Stephen Abok");
                System.out.println("version: 0.01");
            }
            
            //Print the current time
            if (commandSplit[0].equals("time")) {
                command.add(commandLine); //Add input into array
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat formatter = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(formatter.format(time));
            }
            
            //Print today's date
            if (commandSplit[0].equals("date")) {
                command.add(commandLine); //Add input into array
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat formatter = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(formatter.format(date));
            }
            
            //Exit the console 
            if (commandSplit[0].equals("exit")) {
                command.add(commandLine); //Add input into array
                System.exit(0);
            }
            
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error: Not enough input argument(s). Please enter hist -h for help. ");
                }
                else {
                    //Print help list
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                    }
                    //To clear the command history
                    if (commandParams.get(0).equals("-c")) { //When -c is first parameter
                        System.out.println("Deleted: All command hsitory ");
                        
command.clear(); //Clear array
                    }    
                    //Print the history of user inputs
                    else if (commandParams.get(0).equals("-l")) { //When -l is first parameter
                        int parameter = Integer.parseInt((String) commandParams.get(1)); //To change String to Integer  
                        
                        //To ensure hist -c works
                        if(parameter > 0) {
                            if(command.isEmpty()) {
                                continue;
                            
}
                            else for(int j = 0; j < parameter; j++) {
                                    System.out.println(command.get(j)); //Print command history
                            }
                        }
                        
                        //To ensure hist -c works
                        if(parameter < 0) {
                            if(command.isEmpty()) {
                                continue;
                            }
                            else {
                                System.out.println("Invalid parameter ");
                            }
                        }
                        
                        
if(commandParams.size() > 2) {
                            if (commandParams.get(2).equals("-c")) {
                                System.out.println("Deleted: All command history ");
                                command.clear(); //Clear array
                            }
                        }  
                    }
                }
                    
            }

            //List files and/or directories
            if(commandSplit[0].equals("ls")) {
                
                String dirPath = "C:\\directory";
                String newDirName = "test_folder";
                String newFileName = "test_file.txt";
                
                //Create new directory
                File directory = new File(dirPath);
                boolean isCreated = directory.mkdirs();
                
                //Create new file
                File newFile =  new File(dirPath + File.separator + newFileName);
                isCreated = newFile.createNewFile();
                
                //Create new folder(sub-directory)
                File newDirectory = new File(dirPath + File.separator + newDirName);
                isCreated = newDirectory.mkdir();
                
                //display all items in directory
                command.add(commandLine); //Add input into array
                if(commandParams.isEmpty()) {
                    String[] files = directory.list();
                    for(String string : files) {
                        System.out.println(string);
                    }
                }
                else {
                    //display files only
                    if(commandParams.get(0).equals("-f")) {
                        
                        
File[] listOfFiles = directory.listFiles();
                        for(int i = 0; i < listOfFiles.length; i++) {
                            if(listOfFiles[i].isFile()) {
                                System.out.println(listOfFiles[i].getName());
                            }
                        }
                        
                        //display folders(sub-directories) only    
                        if(commandParams.get(0).equals("-d")) {
                        
                            
File[] listOfDirectories = directory.listFiles();
                            for(int i= 0; i < listOfDirectories.length; i ++) {
                                if(listOfDirectories[i].isDirectory()) {
                                System.out.println(listOfDirectories[i].getName());
                                }
                            }
                        }
                    }
                }
            }

            //Launch notepad
            ProcessBuilder process = new ProcessBuilder("Notepad.exe");
            
            if(commandSplit[0].equals("notepad")) {
                
command.add(commandLine); //Add input into array
                process.start();
            }
            
            //When user enter invalid command
            boolean flag = false;
            for(int i = 0; i < cmd.size(); i++) {
                if(commandSplit[0].equals(cmd.get(i))) {
                    flag = true;
                }
            }
           
            if(flag == false) { 
                    System.out.println("Invalid command "); //display error message      
            }
            
        }
    }
}
