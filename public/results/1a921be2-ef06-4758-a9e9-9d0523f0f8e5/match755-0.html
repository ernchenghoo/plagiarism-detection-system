import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;
import java.util.List;

public class myshell_00000171 {

    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("My Shell Version 0.01");
		ArrayList commandHist = new ArrayList();

        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000171> ");

            //Read the input from user.
            commandLine = console.readLine();

            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");

            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }

            //Print the current time.
            else if (commandSplit[0].equals("time")) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));

                commandHist.add(commandSplit[0]);
            }

            //Print today's date.
            else if (commandSplit[0].equals("date")) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));

                commandHist.add(commandSplit[0]);
                //System.out.println(commandHist.get(0));

            else if (commandSplit[0].equals("about")) {
                System.out.println("author: Goh Yi Ren");
                System.out.println("version: 0.01");

                commandHist.add(commandSplit[0]);
                //System.out.println(commandHist.get(1));
            }

            //Exit the console when "exit" is entered.
            else if (commandSplit[0].equals("exit")) {
                System.exit(0);
            }

            //Launch notepad when "notepad" is entered.
            else if (commandSplit[0].equals("notepad")) {
                ProcessBuilder pb = new ProcessBuilder("Notepad.exe");
				pb.start();

				commandHist.add(commandSplit[0]);
            }

            //To retrieve files and folders (or directories) from the current directory
            else if (commandSplit[0].equals("ls")) {
            	if (commandParams.size() < 1) {
					File folder = new File(".");
					File[] listOfFiles = folder.listFiles();

					for (File file : listOfFiles) {
					    if (file.isFile()) {
					        System.out.println(file.getName());
					    }
					    if (file.isDirectory()) {
					        System.out.println(file.getName());
					    }
					}
					commandHist.add(commandSplit[0]);
            	}
            	else if (commandParams.size() == 1)
            	{
            		if (commandParams.get(0).equals("-f"))
            		{
            			File folder = new File(".");
						File[] listOfFiles = folder.listFiles();

						for (File file : listOfFiles) {
						    if (file.isFile()) {
						        System.out.println(file.getName());
						    }
						}
						String command = commandSplit[0];
						commandHist.add(command + " -f");
            		}
            		else if (commandParams.get(0).equals("-d"))
            		{
						File folder = new File(".");
						File[] listOfFiles = folder.listFiles();

						for (File file : listOfFiles) {
						    if (file.isDirectory()) {
						        System.out.println(file.getName());
						    }
						}
						String command = commandSplit[0];
						commandHist.add(command + " -d");
            		}
            		else
            		{
            			System.out.println("Invalid command.");
            		}
            	}
            	else
            	{
            		System.out.println("Invalid command.");
            	}
            }

            else if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } else {
                    //Process based on the number of parameters attached to the command.
                    if (commandParams.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                    }

                    else if (commandParams.get(0).equals("-c")) {
                    	if (commandParams.size() == 1)
                    	{
                    		commandHist.clear();
                    	}
                    	else if(commandParams.size() == 2)
                    	{
                    		if (commandParams.get(1).equals("-l"))
                    		{
								System.out.println("Error: Not enough input argument(s).");
                    		}
                    		else
                    		{
                    			System.out.println("Error: Invalid command.");
                    		}
                    	}
                    	else if(commandParams.size() == 3)
                    	{
                    		if (commandParams.get(1).equals("-l"))
                    		{
								String strLookBack2 = String.join(",", commandParams);
	        					String lookBack2 = strLookBack2.substring(strLookBack2.lastIndexOf(',') + 1).trim();

	        					try {
								    int intLookBack2 = Integer.parseInt(lookBack2);
								    int intLoopBack2 = intLookBack2;

									commandHist.clear();
								} catch (NumberFormatException e) {
							    	System.out.println("Error: Received input that is not an integer.");
								}
                    		}
                    		else
                    		{
                    			System.out.println("Error: Invalid command.");
                    		}
                    	}
                    	else
                    	{
                    		System.out.println("Error: Invalid command.");
                    	}
                    }

                 	else if (commandParams.get(0).equals("-l")) {
                 		if (commandParams.size() == 2)
                 		{
                 			String strLookBack = String.join(",", commandParams);
	        				String lookBack = strLookBack.substring(strLookBack.lastIndexOf(',') + 1).trim();

	        				//int intLookBack = Integer.parseInt(lookBack);
	        				//int intLoopBack = intLookBack;

	        				try {
							    int intLookBack = Integer.parseInt(lookBack);
							    int intLoopBack = intLookBack;
							    if (intLookBack > 0 && intLookBack > commandHist.size())
								{
									System.out.println("Error: Input number larger than commands in command history. Current amount commands in history: " + commandHist.size());
								}

								else if (intLookBack > 0 && intLookBack <= commandHist.size())
								{
									while (intLoopBack > 0)
									{
		        						int size = commandHist.size();
		        						int startLoop = commandHist.size() - intLoopBack;
										System.out.println(commandHist.get(startLoop));
										intLoopBack = intLoopBack - 1;
									}
									String command = "hist -l " + intLookBack;
									commandHist.add(command);
								}
								else
								{
									System.out.println("Error: Invalid number entered.");
								}
							} catch (NumberFormatException e) {
							    System.out.println("Error: Received input that is not an integer.");
							}
                 		}
                 		else if (commandParams.size() < 2)
                 		{
                 			System.out.println("Error : Not enough input argument(s).");
                 		}
                 		else if (commandParams.size() > 3)
                 		{
                 			System.out.println("Error: Invalid command.");
                 		}
                 		else
                 		{
                 			if (commandParams.get(2).equals("-c"))
                 			{
                 				String strLookBack1 = String.join(",", commandParams);
	        					String trimmed = strLookBack1.substring(strLookBack1.indexOf(',') + 1).trim();
								String lookBack1 = trimmed.substring( 0, trimmed.indexOf(","));

		        				//int intLookBack1 = Integer.parseInt(lookBack1);
		        				//int intLoopBack1 = intLookBack1;
								try {
								    int intLookBack1 = Integer.parseInt(lookBack1);
								    int intLoopBack1 = intLookBack1;

								    if (intLookBack1 > 0 && intLookBack1 > commandHist.size())
									{
										System.out.println("Error: Input number larger than commands in command history. Current amount commands in history: " + commandHist.size());
									}

									else if (intLookBack1 > 0 && intLookBack1 <= commandHist.size())
									{
										while (intLoopBack1 > 0)
										{
			        						int size = commandHist.size();
			        						int startLoop1 = commandHist.size() - intLoopBack1;
											System.out.println(commandHist.get(startLoop1));
											intLoopBack1 = intLoopBack1 - 1;
										}
										String command = "hist -l " + intLookBack1;
										commandHist.add(command);
									}
									else
									{
										System.out.println("Error: Invalid number entered.");
									}
	                 				commandHist.clear();
								} catch (NumberFormatException e) {
								    System.out.println("Error: Received input that is not an integer.");
								}
                 			}
                 			else
                 			{
                 				System.out.println("Error: Invalid command.");
                 			}
                 		}
                    }

                    else
                    {
                    	System.out.println("Error: Invalid command.");
                    }
                }
            }

            else
            {
            	System.out.println("Error: Invalid command.");
            }
        }
    }
}
