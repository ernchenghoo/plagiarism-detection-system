import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.util.Iterator;
import java.text.*;

public class myshell_00000110 {
    
    public static void main (String[] args) throws IOException {
        
        String commandLine;
        
        BufferedReader console = new BufferedReader (new InputStreamReader(System.in));
        
        System.out.println("My Shell Version 0.01");
        
        while (true){
            
            System.out.println("00000110> ");
            
            commandLine = console.readLine();
            
            ArrayList commandAll = new ArrayList();
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            if (commandSplit.length > 1){
                for (int i = 1; i < commandSplit.length; i++){
                    commandParams.add(commandSplit[i]);
                }
            }
            
            if (commandSplit[0].equals("")) {
                continue;
            }
            if (commandSplit[0].equals("about")) {
                commandAll.add(commandLine);
                System.out.println("author: Siah Ti Sean");
                System.out.println("version: 0.01");
            }
            
            else if (commandSplit[0].equals("time")){
                commandAll.add(commandLine);
                Date time = new Date();
                
                SimpleDateFormat systime = new SimpleDateFormat("HH:MM:SS A");
                System.out.println(systime.format(time));
            }
            
            else if (commandSplit[0].equals("date")){
                commandAll.add(commandLine);
                Date date = new Date();
                
                SimpleDateFormat sysdate = new SimpleDateFormat(" E DD/MM/YYYY");
                System.out.println(sysdate.format(date));
            }
            
            else if (commandSplit[0].equals("exit")){
                System.exit(0);
            }
            
            if (commandSplit[0].equals("ls")){
                File file = new File(".");
                String [] fds;
                fds = file.list();
                if(commandParams.size()>0){
                    if(commandSplit[1].equals("-f")){
                        commandAll.add(commandLine);
                        for(String fd : fds){
                            if (fd.contains(".")){
                                System.out.println(fd);
                            }
                        }
                    }
                    if(commandSplit[1].equals("-d")){
                        commandAll.add(commandLine);
                        for(String fd : fds){
                            if(fd.contains(".")){
                                continue;
                            }
                            System.out.println(fd);
                        }
                    }
                } else{
                    commandAll.add(commandLine);
                    for(String fd : fds) {
                        System.out.println(fd);
                    }
                }
            }
            
            else if (commandSplit[0].equals("notepad")) {
                commandAll.add(commandLine);
                ProcessBuilder proc = new ProcessBuilder("notepad.exe");
                proc.start();
            }
            
            if (commandSplit[0].equals("hist")){
                if (commandParams.size() <1){
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } else {
                    if (commandParams.get(0).equals("-h")){
                        System.out.println("hist -1 [int]: Retrieve the last [int] commands entered by the user");
                        System.out.println("hist -c : Clear the commandn history.");
                    }
                    else if (commandParams.get(0).equals("-c")){
                        System.out.println("Command history cleared");
                        commandAll.clear();
                        if(commandParams.get(1).equals("-1")){
                            int x = Integer.parseInt(commandParams.get(2).toString());
                            if (x<0){
                                System.out.println("Invalid number");
                            }
                            else if(x >= commandAll.size()){
                        }
                            else {
                                int a = commandAll.size() - Integer.parseInt(commandParams.get(2).toString()) - 1;
                                int b = commandAll.size() - 1;
                                ArrayList<String> part = new ArrayList<>(commandAll.subList(a, b));
                                Iterator iterator = part.iterator();
                                while(iterator.hasNext()){
                                    System.out.println(iterator.next());
                                }
                            }
                    }
                }
                    else if (commandParams.get(0).equals("-1")){
                        if(commandParams.size()<2){
                            System.out.println("Invalid command");
                        }
                        else{
                            commandAll.add(commandLine);
                            int x = Integer. parseInt(commandParams.get(1).toString());
                            if(x < 0){
                                System.out.println("Invalid number");
                            }
                            else if(x > commandAll.size()){
                                
                            }
                            else {
                                int a = commandAll.size() - Integer.parseInt(commandParams.get(1).toString()) - 1;
                                int b = commandAll.size() - 1;
                                ArrayList<String> part = new ArrayList<>(commandAll.subList(a, b));
                                Iterator iterator2 = part.iterator();
                                while(iterator2.hasNext()){
                                    System.out.println(iterator2.next());
                                }
                            }
                            if(commandParams.size()>2){
                                if(commandParams.get(2).equals("-c")){
                                    System.out.println("Command history cleared");
                                    commandAll.clear();
                                }
                            }
                        }
                    }
                    else{
                        System.out.println("Invalid command");
                    }
            }
        }
            else{
                System.out.println("Invalid command");
            }
        
        }
    }
}
