import java.io.*;

import java.util.ArrayList;
import java.util.Date;
import java.text.*;
import java.util.LinkedList;
import java.util.List;




public class myshell_00000138 {
    
    public static void main(String[] args) throws IOException{

        String commandLine;
        
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        ArrayList<String> commandHist=new ArrayList<String>();
        
        File f=new File(".");
        File[] files= f.listFiles();




        while (true) {

            System.out.print("00000138> ");

            commandLine = console.readLine();

            
            ArrayList<String> commandParams = new ArrayList<String>();
            String[] commandSplit = commandLine.split(" ");
            
            if(commandSplit[0].equals("about") ||
                   commandSplit[0].equals("time") ||
                       commandSplit[0].equals("date")||
                          commandSplit[0].equals("hist")||
                              commandSplit[0].equals("ls")||
                                  commandSplit[0].equals("notepad")||
                                      commandSplit[0].equals("exit")){

                if (commandSplit.length > 1) {
                    for (int i = 1; i< commandSplit.length; i++) {
                         commandParams.add(commandSplit[i]);                   
                        }
                    }    


                if (commandSplit[0].equals("")) {
                    continue;
                }
                else{
                    commandHist.add(commandSplit[0]);
                }

                 //ABOUT
                if(commandSplit[0].equals("about")){
                    System.out.println("author: Muhammad Irfaan Ismael Issany");
                    System.out.println("version: 0.01");
                }

                //TIME
                if (commandSplit[0].equals("time")) {
                    Date time = new Date();
                    SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                    System.out.println(sdf.format(time));
                }

                //DATE
                if (commandSplit[0].equals("date")) {
                    Date date = new Date();
                    SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                    System.out.println(sdf.format(date));
                }

                //Exit
                if (commandSplit[0].equals("exit")) {
                    System.exit(0);
                }



                
                if (commandSplit[0].equals("ls")) {
                    if(commandParams.size()<1){
                        for(File F:files){
                            System.out.println(F.getName());
                        }
                    }
                    
                    else if(commandParams.get(0).equals("-f"))
                    {
                        for(File F:files)
                        {
                            if(F.isFile())
                            {
                                System.out.println(F.getName());
                            }
                        }
                    }
                    else if(commandParams.get(0).equals("-d"))
                    {
                        for(File F:files)
                        {
                            if(F.isDirectory())
                            {
                                System.out.println(F.getName());
                            }
                        }
                    }
                }
                else if(commandSplit[0].equals("notepad"))
                {
                    ProcessBuilder Np=  new ProcessBuilder("notepad.exe");
                    Np.start();
                }

                
                //HISTORY
                if (commandSplit[0].equals("hist")) {
                    if (commandParams.size() < 1) {
                        System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    } else {
                            for(int x=0; x<commandParams.size();x++){
                        
                        if (commandParams.get(x).equals("-h")) {
                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                            System.out.println("hist -c : Clear the command history.");
                        }

                        if (commandParams.get(x).equals("-c")) {
                           
                            int z=commandHist.size();
                            for(int a=0; a<z; a++){
                                commandHist.set(a, "");
                            }
                        }
                        if(commandParams.get(x).equals("-l")){
                            int a= Integer.parseInt(commandSplit[x+2]);
                            if(a>0){
                                
                                    for( int y=a; y>=1 ;y--){
                                    System.out.println(commandHist.get(commandHist.size()-1-y)+ " ");


                             }       
                            }
                        else{
                            
                                System.out.println("Invalid number");
                        }     

                    } 
                    
                }
             }
         
          }
      }
            else{
                System.out.println("Invalid command!");
            }
        }
        
    }
}
