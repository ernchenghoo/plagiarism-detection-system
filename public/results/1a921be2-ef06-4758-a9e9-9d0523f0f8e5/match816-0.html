import java.io.BufferedReader;
import java.io.File;
import java.io.*;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000143 {
     public static void main(String[] args) throws IOException  {
       String commandLine;
        
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
       
        System.out.println("my shell version 0.01");
        
        ArrayList <String> command= new ArrayList<>();
     
      
        
        ProcessBuilder pb= new ProcessBuilder("notepad.exe"); 
        
          while(true){
            System.out.printf("00000143> ");
            commandLine= console.readLine();
            
           ArrayList<String> commandParams = new ArrayList<>();
           String[] commandSplit = commandLine.split(" ");
           
           if (commandSplit[0].equals("about") || commandSplit[0].equals("time") || commandSplit[0].equals("date") || commandSplit[0].equals("exit")
        	|| commandSplit[0].equals("hist") || commandSplit[0].equals("ls") || commandSplit[0].equals("notepad")) { 
	            
	            if(commandSplit.length>1){
	                for(int i=1; i<commandSplit.length; i++){
	                    commandParams.add(commandSplit[i]);
	                }
	            }
	            
	            if (commandSplit[0].equals("")) {
	                continue;
	            }
	            else{
	                command.add(commandSplit[0]);
	            }
	            
	            if(commandSplit[0].equals("about")){
	                System.out.println("author: Bibi Aamirah Shafaa Emambocus \nversion: 0.01");
	            }
	            
	            if (commandSplit[0].equals("time")) {
	                Date time = new Date();
	                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
	                System.out.println(sdf.format(time));
	            }
	            
	            if (commandSplit[0].equals("date")) {
	                Date date = new Date();
	                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
	                System.out.println(sdf.format(date));
	            }
	            
	            if (commandSplit[0].equals("exit")) {
	                System.exit(0);
	            }
	            
	            if (commandSplit[0].equals("notepad")) {
	            	pb.start();
	            }
	            
	            if (commandSplit[0].equals("ls")) {
	               File f = new File(".");
	               File[] files= f.listFiles();
	                 
	                if(commandParams.isEmpty()) {
	                	for (File F:files) {
	                		System.out.println(F.getName());
	                	}
	                }
	                else {
		                if(commandParams.get(0).equals("-f")) {
		                	for(File F:files) {
		                		if(F.isFile()) { System.out.println(F.getName());}
		                	}
		                }
		                if(commandParams.get(0).equals("-d")) {
		                	for(File F:files) {
		                		if(F.isDirectory()) { System.out.println(F.getName());}
		                	}
		                }
	                }
	                
	            } 
	            
	            if (commandSplit[0].equals("hist")){
	            	 if(commandParams.isEmpty()){ 
	                        System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
	                    }
	                 else{
	            	
	                	 for (int c=0; c<commandParams.size(); c++){
	                		 
	                      if (commandParams.get(c).equals("-h")){
	                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
	                            System.out.println("hist -c : Clear the command history.");
	                        }
	                        if (commandParams.get(c).equals("-l")){
	                            int x= Integer.parseInt(commandSplit[c+2]);
	                            if(x<0){
	                                System.out.println("Invalid number");
	                            }
	                            else{
	                                for(int i=x; i >= 1; i--){
	                                    System.out.println(command.get(command.size()-1-i)+ " ");
	                                   
	                                }
	                            }
	                        }
	
	                        if (commandParams.get(c).equals("-c")){
	                            System.out.println("Deleted: All command history.");
	                            int x= command.size();
	                            for (int i=0; i<x; i++){
	                                command.set(i,"");
	                            }
	                            
	                        }
	
	                    }
	                }
                         
	            }
	            
	        
           } 
           else {
        	   System.out.println("Invalid command!");
           }
        }
        
    }
}
