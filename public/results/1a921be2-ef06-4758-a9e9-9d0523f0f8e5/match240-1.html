package myshell;

import java.io.*;
import java.util.*;
import java.text.*;

public class myshell_00000068 {

	public static void main(String[] args) throws IOException {
        
		String commandLine;
		BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        
		List<String> history = new ArrayList<String>();
		
		System.out.println("My Shell Version 0.01");
				
		while (true) {
			
			System.out.print("00000068> ");
            
			commandLine = console.readLine();
			
			ArrayList commandParams = new ArrayList();
			String[] commandSplit = commandLine.split(" ");
			
			if(commandSplit.length>1) {
				for(int i = 1; i < commandSplit.length; i++) {
					commandParams.add(commandSplit[i]);
				}
			}
			
			if (commandSplit[0].equals("")) {
				continue;
			}
			
			else if (commandSplit[0].equals("time")) {
				Date time = new Date();
				SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
				System.out.println(sdf.format(time));
				history.add(commandLine);
			}
			
			else if (commandSplit[0].equals("date")) {
				Date date = new Date();
				SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
				System.out.println(sdf.format(date));
				history.add(commandLine);

			}
			
			else if (commandSplit[0].equals("exit")) {
				System.exit(0);
			}

//------------------------------------------------------------------------
			
			//Displays the author and the version number
			else if (commandSplit[0].equals("about")) {
				System.out.println("author: Amanda Tan Joy Mun");
				System.out.println("version:
 0.01");
				history.add(commandLine);
			}

//------------------------------------------------------------------------

			//Create a process to launch notepad
			else if (commandSplit[0].equals("notepad")) {
				ProcessBuilder pb = new ProcessBuilder();
				pb.command("notepad.exe");
				pb.start();
				history.add(commandLine);
			}
			
//------------------------------------------------------------------------
			
			//List both files and directory
			else if (commandSplit[0].equals("ls")) {
				
				File folder = new File(".");
				File[] listOfFiles = folder.listFiles();
				
				if(commandParams.size()<1) {
					for (int j=0; j<listOfFiles.length;
 j++) {
						if(listOfFiles[j].isFile()) {
							System.out.println(listOfFiles[j].getName());
						} else if(listOfFiles[j].isDirectory()) {
							System.out.println(listOfFiles[j].getName());
						}
					} history.add(commandLine);
				} 
				else {
					//List files
					if (commandParams.get(0).equals("-f")) {
						for (int j=0; j<listOfFiles.length; j++) {

							if(listOfFiles[j].isFile()) {
								System.out.println(listOfFiles[j].getName());
							}
						} history.add(commandLine);
					}
					//List directory
					if (commandParams.get(0).equals("-d")){
						for (int j=0; j<listOfFiles.length; j++) {

							if(listOfFiles[j].isFile()) {
								System.out.println(listOfFiles[j].getName());
							}
						} history.add(commandLine);
					}
				}
			}
			
//------------------------------------------------------------------------
			
			else if (commandSplit[0].equals("hist")) {
				if (commandParams.size() < 1) {
					System.out.println("Error. Not enough input argument(s). Please enter -h for help.");
					history.add(commandLine);	
				} 
				
				else {
					for(int i = 0; i < commandParams.size(); i++) {
						if (commandParams.get(i).equals("-h")) {
							System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
							System.out.println("hist -c : Clear the command history.");
							history.add(commandLine);
						}
					
					//Clear the command history
					if (commandParams.get(i).equals("-c")) {
						history.clear();
					}
					
					//Prints previous x number of used commands
					if (commandParams.get(i).equals("-l")) {
						if (commandParams.size()<2) {
							for (i = 0; i < history.size(); i++) {
								System.out.println(history.get(i));
							} history.add(commandLine);
						}
						
						else if (commandParams.size()<3) {
							if (history.isEmpty()) {
								history.add(commandLine);
								break;
							}
								else {
									if(history.size() < Integer.parseInt((String) commandParams.get(1))) {
										System.out.println("Not enough commands");
										break;
									}
									else {
										for (i = Integer.parseInt((String) commandParams.get(1)); i > 0; i--) {
											System.out.println(history.get(history.size()-i));
										} history.add(commandLine);
									} 
									
								}
							
							}
						}
					}
				}
			}
			else {
				System.out.println("Invalid command");
				history.add(commandLine);
			}
		}
	}
}
