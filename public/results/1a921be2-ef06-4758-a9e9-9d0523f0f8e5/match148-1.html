import java.io.BufferedReader;
import java.io.File;
import java.io.IOException;
import java.io.InputStreamReader;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Date;

public class myshell_00000007 {
    
    public static void main(String[] args) throws IOException {
        
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        ArrayList<String> userCommands = new ArrayList();
        
        while(true){
            
            System.out.print("00000007> ");
            String commandLine = br.readLine();
            ArrayList<String> commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            if(commandSplit[0].contains("about") || commandSplit[0].contains("time") || commandSplit[0].contains("date") || commandSplit[0].contains("hist") || commandSplit[0].contains("ls") || commandSplit[0].contains("notepad") || commandSplit[0].contains("exit")){
                userCommands.add(commandLine);
                if(commandSplit.length > 1){
                    for(int i = 1; i < commandSplit.length; i++){
                        commandParams.add(commandSplit[i]);
                    }
                }
            }
           
            // Display author and version number (0.01).
            if(commandSplit[0].equals("about")){
                
                System.out.println("author: Thay An Gie");
                System.out.println("version: 0.01");
            }
            
            // Display time.
            if(commandSplit[0].equals("time")){
                Date time = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
            }
            
            // Display date.
            if(commandSplit[0].equals("date")){
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
            }
            
            if(commandSplit[0].equals("hist")){
                if(commandParams.size() < 1){
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } else {
                    if(commandParams.get(0).equals("-h")){
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                    }
                    if(commandParams.get(0).equals("-l")){
                        int userInputValue = Integer.parseInt(commandParams.get(1));
                            // If x is not a valid number.
                            if(userInputValue < 0){
                                System.out.println("Invalid number");
                            }
                            // Display previous x number of used commands.
                            if(userInputValue < userCommands.size()){
                                for(int i = 0; i < userInputValue; i++){
                                    System.out.println(userCommands.get(i));
                                
}  
                                // Print previous x number of used commands and then clear the command history.	
                                if(commandParams.size() > 2){
                                    if(commandParams.get(2).equals("-c")){
                                        System.out.println("Cleared all command history");
                                        userCommands.clear();
                                    }
                                }
                            }
                    }
                    // Clear command history.
                    if(commandParams.get(0).equals("-c")){
                        System.out.println("Deleted: All command history.");
                        userCommands.clear();
                    }
                }
            }
            
            // List all files and directories in the current directory.
            if(commandSplit[0].equals("ls")){
                File directory = new File("C:\\Program Files\\NetBeans 8.2");
                File[] files = directory.listFiles();
                if(commandParams.size() < 1){
                    for(File file: files){
                        System.out.println(file);
                    }
                } else 
{
                    // List only files in the current directory.
                    if(commandParams.get(0).equals("-f")){
                        for(File file: files){
                            if((file.isFile())){
                                
System.out.println(file);
                            }
                        }
                    }
                    // List only directories in the current directory.
                    if(commandParams.get(0).equals("-d")){
                        for(File file: files){
                            if((file.isDirectory())){
                                System.out.println(file);
                            }
                        }
                    }
                
}
            }
            
            // Create a process to launch notepad.
            if(commandSplit[0].equals("notepad")){
                ProcessBuilder pb = new ProcessBuilder("Notepad.exe");
                pb.start();
            }
            
            // Exit the program.
            if(commandSplit[0].equals("exit"))
{
                System.exit(0);
            }  
            
            // Validate user command.
            if(!(commandSplit[0].contains("about") || commandSplit[0].contains("time") || commandSplit[0].contains("date") || commandSplit[0].contains("hist") || commandSplit[0].contains("ls") || commandSplit[0].contains("notepad") || commandSplit[0].contains("exit"))){
                System.out.println("Invalid command");
            }
        }
    }
}
