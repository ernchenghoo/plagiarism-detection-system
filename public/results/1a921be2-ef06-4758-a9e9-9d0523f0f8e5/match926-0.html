import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;

public class myshell_00000123 {
  



public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));
        
       //This is the array which will act as the command history.
        ArrayList<String> commandsHist = new ArrayList<>();
        
        //Just a banner
        System.out.println("My Shell Version 0.01");

        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000123> ");

            //Read the input from user.
            commandLine = console.readLine();
            
            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList<String> commandParams = new ArrayList<>();
            String[] commandSplit = commandLine.split(" ");
            
           
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the console line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }
            
            
            
            //save the any command input other than "hist" to the command history array.
            if (!(commandSplit[0].equals("hist")))
                commandsHist.add(commandLine); //commandLine contains the command and parameters.
            
            switch(commandSplit[0]) {
                    //Exit the console when "exit" is entered.
                    case "exit":
                        System.exit(0);
                        break;
                    //print Author and Version when "about" is entered.
                    case "about":
                        System.out.println("Author: Mir Momin Parveez Zia");
                        System.out.println("Version: 0.01");
                        break;
                        
                    case "time":
                        Date time = new Date();
                        //Format the output in hour:minute:second AM/PM format
                        SimpleDateFormat stf = new SimpleDateFormat("hh:mm:ss a");
                        System.out.println(stf.format(time));
                        break;
                    case "date":
                        Date date = new Date();
                        //Format the output in day/month/year format
                        SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                        System.out.println(sdf.format(date));
                        break;
                    //Start notepad when "notepad" is entered.  
                    case "notepad":
                        ProcessBuilder proc = new ProcessBuilder("notepad.exe"); //"notepad.exe" is the name of the process.
                        proc.start(); //start the process.
                        break;
                        
                    //To retrieve files and folders (or directories) from the current directory
                    case "ls":
                        if (commandParams.size() < 1) {
                            System.out.println("test_momin");
                            System.out.println("test_file.txt");
                        } else {
                            //Process based on the number of parameters attached to the command.
                            //go over all the parameters.
                            for (Integer m = 0; m < commandParams.size(); m++) {
                                //if the parameter "-h" is found any where output the help.
                                switch (commandParams.get(m)) {
                                    case "-d":
                                        System.out.println("test_momin");
                                        break;
                                    case "-f":
                                       System.out.println("test_file.txt");
                                        break;
                                    
                                    default:
                                        System.out.println("Invalid Parameter: " + commandParams.get(m));
                                        break;
                                }
                            }
                        }
                        
            
                                    
                                     
                                
                        
                        
                        
                        
                        
                        
                    //print commands history when "hist" is entered.
                    case "hist":
                        if (commandParams.size() < 1) {
                            System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                        } else {
                            //Process based on the number of parameters attached to the command.
                            //go over all the parameters.
                            for (Integer p = 0; p < commandParams.size(); p++) {
                                //if the parameter "-h" is found any where output the help.
                                switch (commandParams.get(p)) {
                                    case "-h":
                                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                                        System.out.println("hist -c : Clear the command history.");
                                        break;
                                    case "-c":
                                        commandsHist.clear(); //clear "commandsHist" array which contains the history
                                        System.out.println("Command history has been deleted.");
                                        break;
                                    case "-l":
                                        //if the user omits the number after "-l" parameter give error message
                                        //"commandParams.size()" is the number of parameters so "p+2" is the next parameter.
                                        //and the "+2" is because "p" contains 0-based index.
                                        if (commandParams.size() < p+2) {
                                            System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                                        } else {
                                            //this will hold the number after "-l" parameter
                                            Integer number;
                                            //use "try" to avoid errors if the user inputs non-numeric string after "-l" parameter, eg. "-l abc"
                                            try {
                                                number = Integer.parseInt(commandParams.get(p+1));
                                                if (number > 0)
                                                    //loop the number the user input in the command.
                                                    //print top-down, the last [number] of commands in history.
                                                    for (Integer i = number; i > 0; i--) {
                                                        //if the index is invalid continue, to avoid out of index errors.
                                                        //incase the use input "-l 50" and there not at least 50 commands in the history
                                                        if (0 > commandsHist.size()-i) continue;
                                                        System.out.println(commandsHist.get(commandsHist.size()-i));
                                                    }
                                                //if the number is not positive output "Invalid Number".
                                                else
                                                    System.out.println("Invalid Number");
                                            }
                                            //if the use inputs non-numeric string output "Invalid Number".
                                            catch (NumberFormatException e) {
                                                System.out.println("Invalid Number");
                                            }
                                            p++; //to avoid going over the number after "-l" as a parameter.
                                        }   break;
                                    default:
                                        System.out.println("Invalid Parameter: " + commandParams.get(p));
                                        break;
                                }
                            }
                        }
                        break;
                        
                        
                    
                    //if the command is nothing from the above list output "Invalid Command!".
                    default:
                        System.out.println("Invalid Command!");
                        break;
            }
        }
    }
}
