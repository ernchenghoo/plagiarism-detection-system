	    public static void HistoryLast(ArrayList<String> history , int y) {
	    	
	    	//initialize string variable to null
	    	String value="";
	    
	    	//set boolean variable to true
	        boolean booHist = true;
	       
	        for (int x = y; x < history.size(); x++) {
	            if (history.get(x).equals("-l")) {
	                x++;

	                if (booHist) {
	                	
	                    if (x < history.size()) {
	                        String outcome = Hist(history, history.get(x).toString());
	                        
	                        if (value.equals("Invalid Number")) {
	                            value += outcome + "\n";
	                            x--;
	                        }
	                         
	                        else {
	                            value += outcome + "\n";
	                        }
	                    }
	                    else {
	                    	//this message will pop up if no number have been inserted 
	                        value += "\nInvalid Number";
	                    }
	                }
	                
	            }
	            
	        }

	        System.out.println(value.substring(0, value.length() - 1));
	    }
 
	    public static String Hist(ArrayList<String> history, String value)
	    {
	    	//initialize string variable to null
	    	String outcome = "";
	    	int values;
	        
	        try {
	        	
	            values = Integer.parseInt(value);

				int hists = hist.size();

	            for (int x = 0; x < values; x++) {
	               
	                hists--;
	                outcome += hist.get(hists).toString() + "\n";
	            }

	            return outcome.substring(0, outcome.length() - 1);
	            
	        } catch (Exception ex) {
	        	
	        	//this message will pop up if number does not match the history
	            return "\nInvalid Number";
	        }
