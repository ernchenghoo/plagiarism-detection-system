import java.io.*;
import java.util.*;
import java.text.*;

public class myshell_00000057 {

    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;
        
        //Use commands to store commands
        ArrayList commands = new ArrayList();

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("My Shell Version 0.01");

        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000057> ");

            //Read the input from user.
            commandLine = console.readLine();
            
            //Store commands
            commands.add(commandLine);

            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");

            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }

            //Reprint the console line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            }
            
            outerloop:
            //Display invalid command error if user input unneccesary arguments
            if ((commandSplit[0].equals("time"))||(commandSplit[0].equals("date"))||(commandSplit[0].equals("about"))||(commandSplit[0].equals("notepad"))||(commandSplit[0].equals("exit"))) {
                if (commandParams.size() >= 1) {
                    System.out.println("Invalid Command!");
                }                
                else {
                    //Print the current time.
                    if (commandSplit[0].equals("time")) {
                        Date time = new Date();
                        //Format the output in hour:minute:second AM/PM format
                        SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                        System.out.println(sdf.format(time));
                    }

                    //Print today's date.
                    else if (commandSplit[0].equals("date")) {
                        Date date = new Date();
                        //Format the output in day/month/year format
                        SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                        System.out.println(sdf.format(date));
                    }

                    // Display the author and the version number
                    else if (commandSplit[0].equals("about")){
                        
System.out.println("author: Koh Mei Quen");
                        System.out.println("version: 0.01");
                    }

                    //Create a process to launch notepad
                    else if (commandSplit[0].equals("notepad")) {
                        //Create the process builder
                        ProcessBuilder pb = new ProcessBuilder("notepad.exe");
                        System.out.println("Starting notepad...");
                        //Start the process
                        pb.start();          
                    }

                    //Exit the console when "exit" is entered.
                    else if (commandSplit[0].equals("exit")) {
                        System.exit(0);
                    }
                }
            }
                        
            //To retrieve files and folders (or directories) from the current directory
            else if (commandSplit[0].equals("ls")) {
                File f = null;
                File[] paths;

                // create new file
                f = new File("").getAbsoluteFile();

                // returns pathnames for files and directory
                paths = f.listFiles();

                // for each pathname in pathname array
                for(File path:paths) {
                    
                    //Prints both files and directories if no arguments
                    if (commandParams.size() < 1) {                        
                        // prints file and directory names
                        System.out.println(path.getName());          
                    } 
                    else if (commandParams.size() == 1 ) {
                        //Prints only file names if "-f" is appended
                        if (
commandParams.get(0).equals("-f")) {
                            if (path.isFile()) {
                                System.out.println(path.getName());
                            }
                        } 
                        //Prints only directory names if "-d" is appended
                        else if (commandParams.get(0).equals("-d"))
{
                            if (path.isDirectory()) {
                                System.out.println(path.getName());
                            }
                        } 
                        //Invalid command if argument appended is not "-f" or "-d"
                        else {
                            System.out.println("Invalid Command!");
                            break;
                        }
                    }
                    //Invalid command if argument appended is more than 1
                    else {
                        System.out.println("Invalid Command!");
                        break;
                    }
                }             
            }
            
            //Example of a built-in function that can be used to print the command history. 
            else if (
commandSplit[0].equals("hist")) {
                
                if (commandParams.size() < 1) {
                    //Display error when input argument is not enough
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                }                 
                //If input argument is enough
                else {                    
                    //Set iterator for commandParams
                    Iterator paramIterator = commandParams.iterator();
                    
                    //Check for invalid commands when user enters more than one parameter
                    if (commandParams.size() > 1) {
                        while (paramIterator.hasNext()) {
                            String param = (String) paramIterator.next();
                            if ((param.equals("-h"))||(param.equals("-c"))){
                                continue;
                            }
                            else if ((param.equals("-l"))) {
                                //Initialize int x
                                int x = 0;
                                
                                //Catch Exception if x is not an integer or x is not specified
                                try {
                                    x = Integer.parseInt((String)paramIterator.next());
                                }
                                catch (Exception e){
                                    System.out.println("Invalid Command!");
                                    break outerloop;
                                }
                                
                                //Invalid number (negative number)
                                if (x < 0){
                                    System.out.println("Invalid Number/Command!");
                                    break outerloop;
                                }                                
                                //Valid number (positive number)
                                else{
                                    continue;
                                }
                            }
                            //Invalid command
                            else {
                                System.out.println("Invalid Command!");
                                break outerloop;
                            }
                        }
                    }
                    
                    paramIterator = commandParams.iterator();
                    while (paramIterator.hasNext()) {                        
                        //Assign the next element from paramIterator to param
                        String param = (String) paramIterator.next();
                    
                        //Process based on the number of parameters attached to the command.
                        //Show help for hist commands
                        if (param.equals("-h")) {
                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                            
System.out.println("hist -c : Clear the command history.");
                        }
                        
                        //Delete command history
                        else if (param.equals("-c")) {
                            System.out.println("Deleted: All command history.");
                            commands.clear();
                        }
                        
                        //Print previous x number of used commands
                        else if (param.equals("-l")){
                            
                            //If x is not entered, display error as input argument is not enough
                            if (commandParams.size() < 2) {
                                System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                            }
                            
                            else {                                
                                //Initialize int x
                                int x = 0;
                                
                                //Catch Exception if x is not an integer or x is not specified
                                try {
                                    x = Integer.parseInt((String)paramIterator.next());
                                }
                                catch (Exception e){
                                    System.out.println("Invalid Number");
                                }

                                //Invalid number (negative number)
                                if (x < 0){
                                    System.out.println("Invalid Number");
                                }
                                
                                //Valid number (positive number)
                                else{                                    
                                    //If x is less than the number of used commands, print previous x number of used commands as usual
                                    if (x < commands.size()){
                                        while (x > 0){
                                            System.out.println(commands.get(commands.size()-1-x));
                                            x--;
                                        }
                                    }
                                    
                                    //Uses ListIterator if x is more than the number of used commands to avoid ArrayIndexOutOfBoundsException and print all used commands
                                    else{
                                        ListIterator commandIterator = commands.listIterator();
                                        while (commandIterator.hasNext()){
                                            if (commandIterator.nextIndex() >= commands.size()-1){
                                                break;
                                            }
                                            else{
                                                System.out.println(commandIterator.next());
                                            }
                                        }
                                    }
                                }
                            }
                        }
                        
                        //Display Invalid Command error when user enters a command that is not in the list
                        else {
                            System.out.println("Invalid Command!");
                        
}
                    }
                }
            }
                        
            //Display Invalid Command error when user enters a command that is not in the list
            else {
                System.out.println("Invalid Command!");
            }
        }
    }
}
