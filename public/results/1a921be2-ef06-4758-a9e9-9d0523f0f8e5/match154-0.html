import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;
import java.nio.file.Paths;

public class myshell_00000159 {
     
    public static void main (String[] args) throws IOException{
        String commandLine;
        BufferedReader console = new BufferedReader (new InputStreamReader(System.in));
        
        System.out.println("My Shell Version 0.01");      
        ArrayList commandHist = new ArrayList();
        
        while (true){
            
            boolean flag = true;
            
            System.out.print("00000159> ");
            commandLine = console.readLine();            
            ArrayList commandParams = new ArrayList();            
            String[] commandSplit = commandLine.split(" ");
            
            if (commandSplit.length > 1){
                for (int i=1; i < commandSplit.length; i++){
                    commandParams.add(commandSplit[i]);
                }
            }
            
            if (commandSplit[0].equals("")){
                continue;               
            }
            
            if (commandSplit[0].equals("about")){
                System.out.println("author: Muhammad Danial");
                System.out.println("version: 0.01");                  
            } 
            
            if (commandSplit[0].equals("date")) {
                Date date = new Date();
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));              
            }
            
            if (commandSplit[0].equals("time")){
                Date date = new Date ();
                SimpleDateFormat sdf = new SimpleDateFormat ("hh:mm:ss a");
                System.out.println(sdf.format(date));                  
            } 
            
            if (commandSplit[0].equals("exit")){
                System.exit(0);
            } 
                        
            if (commandSplit[0].equals("ls")){
                System.out.println(Paths.get(".").toAbsolutePath().normalize().toString());
                File folders = new File(Paths.get(".").toAbsolutePath().normalize().toString());
                File[] fList = folders.listFiles();
                
                if (commandParams.size() > 0){
                    if (commandParams.get(0).equals("-f")){
                        for (File fList1 : fList) {
                            if (fList1.isFile()) {
                                System.out.println(fList1.getName());
                            }
                        }
                    }
                    
                    if (commandParams.get(0).equals("d")){
                        for (File fList1 : fList) {
                            if (fList1.isDirectory()) {
                                System.out.println(fList1.getName());
                            }
                        }
                    }
                }else {
                    for (File fList1 : fList) {
                        System.out.println(fList1.getName());
                    }
                }
            }
            
            if (commandSplit[0].equals("notepad")){                                                               
                ProcessBuilder proc = new ProcessBuilder();
                proc.command("notepad.exe");
                proc.start();               
            }   
            
            if (commandSplit[0].equals("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } 
                else if (commandParams.get(0).equals("-h")) {
                    System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                    System.out.println("hist -c : Clear the command history.");
                } 
                else if (commandParams.get(0).equals("-c")) {
                    commandHist.removeAll(commandHist);                     
                    flag = false;
                } 
                else if (commandParams.get(0).equals("-l")) {

                    int num;

                    try {
                        num = Integer.parseInt(commandParams.get(1).toString()); 
                        if (num < 0) {
                            System.out.println("Invalid number"); 
                        }
                    } catch (NumberFormatException e) {
                        commandHist.add(commandLine); 
                        System.out.println("Invalid command"); 
                        continue;
                    }

                    for (int x = 0; x < num; x++) {                           
                        try {
                            System.out.println(commandHist.get(x));
                        } catch (Exception e) {
                        }
                    }
                    
                    if (commandParams.size() > 2) {
                        if (commandParams.get(2).equals("-c")) { 
                            commandHist.removeAll(commandHist);                                                         
                            flag = false; 
                        }
                    }
                } else {
                    System.out.println("Invalid command");
                }
            }    
            if (flag == true){
                commandHist.add(commandLine);
            }

            if (!commandSplit[0].equals("date")&&!commandSplit[0].equals("time")
                    &&!commandSplit[0].equals("about")&&!commandSplit[0].equals("exit")
                    &&!commandSplit[0].equals("hist")&&!commandSplit[0].equals("notepad")&&!commandSplit[0].equals("ls"))
            {
                System.out.println("Invalid Command!");
            }
        }                                             
    }                                
}
