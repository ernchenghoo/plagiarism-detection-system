import java.io.*;
import java.util.*;
import java.text.*;


public class myshell_00000085 {

    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;
        
        //commandHistory is a string array used to store command histories.
        ArrayList <String> commandHistory = new ArrayList <String> ();
        
        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("                 My Shell Version 0.5");
        System.out.println("");
        System.out.println("                    SHELL COMMANDS ");
        System.out.println("          about - displays author and version.");
        System.out.println("          time - displays current time.");
        System.out.println("          date - displays current date.");
        System.out.println("          hist -l x - displays history of x number of commands.");
        System.out.println("          hist -c - clears command history.");
        System.out.println("          ls - displays files and directories in current directory.");
        System.out.println("          ls -f - displays files in the current directory.");
        System.out.println("          ls -d - displays current directory.");
        System.out.println("          notepad - Launches Notepad.");
        System.out.println("          exit - Exits the program/shell.");
        System.out.println("");
        
        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            
            System.out.print("00000085 > ");

            //Read the input from user.
            commandLine = console.readLine();
            
            //Stores user inputs into the commandHistory array.
            commandHistory.add(commandLine);
            
            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParameters = new ArrayList();
            String[] commandSplit = commandLine.split(" ");

            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParameters.add(commandSplit[i]);
                }
            }

            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            } 
            
            //Prints author and version.
            if (commandSplit[0].equals("about")) {
                System.out.println(" Author: Chong Kai Soong");
                System.out.println(" Version: 0.5");
                System.out.println("");
            }
            
            //Print the current time.
            if (commandSplit[0].equals("time")) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                System.out.println("");
            }
            
            //Print today's date.
            if (commandSplit[0].equals("date")) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                System.out.println("");
            }
            
            //Launches the application notepad.
            if (commandSplit[0].equals("notepad")){
                Runtime rs = Runtime.getRuntime();
                     try {
                         rs.exec("notepad");
                     } 
                     catch (IOException e) {
                         System.out.println(e);
                     }
                     System.out.println("Launching Notepad...");
                     System.out.println("");
            }
            
            //To retrieve files and folders (or directories) from the current directory
            File dir = new File (".");
            File[] fileList = dir.listFiles();
            String currentDirectory = System.getProperty("user.dir");
            
            if (commandSplit[0].equals("ls")) {
                 if (commandParameters.size() < 1) {
                     System.out.println("Current directory and files:");
                     System.out.println("");
                     System.out.println("Current Directory > " + currentDirectory);
                     System.out.println("");
                     System.out.println("Files > ");
                         for (File file : fileList){
                                if (file.isFile()){
                                    System.out.println("        " + file.getName());
                                 }
                         }
                     System.out.println("");
                 } else {      
                    if (commandParameters.get(0).equals("-f")) {
                     System.out.println("Files in current directory:");
                     System.out.println("");
                     System.out.println("Files > ");
                         for (File file : fileList){
                                if (file.isFile()){
                                    System.out.println("        " + file.getName());
                                 }
                         }
                        System.out.println("");
                    }
                    if (commandParameters.get(0).equals("-d")) {
                     System.out.println("Current directory:");
                     System.out.println("");
                     System.out.println(currentDirectory);
                     System.out.println("");
                    }
                   }
            }
            
            
            //Example of a built-in function that can be used to print the command history. 
            if (commandSplit[0].equals("hist")) {
                if (commandParameters.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                    System.out.println("");
                } else {
                    //Process based on the number of parameters attached to the command.
                    if (commandParameters.get(0).equals("-h")) {
                        System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                        System.out.println("hist -c : Clear the command history.");
                        System.out.println("");
                    }
                    if (commandParameters.get(0).equals("-l")){
                        int number = Integer.parseInt((String) commandParameters.get(1));
                            if (commandHistory.size() == 1) {
                                continue;
                        }
                        //x is the number of used commands.
                        if (number > 0) {
                            for (int x = 0; x < number; x++) {
                                if (x == commandHistory.size()) {
                                    break;
                                } 
                                System.out.println(commandHistory.get(x));
                            }
                            System.out.println();
                        } else {
                            System.out.println("Invalid number.");
                            System.out.println();
                        }
                    }
                    
                    //-c clears the command history stored in commanHistory ArrayList.
                    if (commandParameters.get(0).equals("-c")) {
                        commandHistory.clear();
                        System.out.println("All command history is DELETED.");
                        System.out.println("");
                        
                    }
                }
            }
            
            //Exit the console when "exit" is entered.
            if (commandSplit[0].equals("exit")) {
                System.exit(0);
            } 
            
            //switch case statements check for invalid inputs.
            switch (commandSplit[0]) {
                
                default:
                    System.out.println("Invalid command.");
                    System.out.println("");
                    break;
                case "about":
                    break;
                case "time":
                    break;
                case "date":
                    break;
                case "notepad":
                    break;
                case "ls":
                    break;
                case "hist":
                    break;
                
            }  
        }     
    } 
}
