import java.io.*;
import java.util.*;
import java.text.*;

public class myshell_00000074 {
    
    private static ArrayList<String> commandHistory = new ArrayList<String>();
    private static final ArrayList<String> COMMANDLIST = new ArrayList<String>();
    public static void main(String[] args) throws IOException {

        String[] commandList = new String[]{"haha", "cls","about", "hist", "ls", "notepad", "exit", "cmd", "help","date","time"};
        ArrayList<ArrayList<String>> helpList = new ArrayList<>();
        
        ArrayList<String> about = new ArrayList<>();
        ArrayList<String> cmdHelp = new ArrayList<>();
        ArrayList<String> cls = new ArrayList<>();
        ArrayList<String> exit = new ArrayList<>();
        ArrayList<String> help = new ArrayList<>();
        ArrayList<String> hist = new ArrayList<>();
        ArrayList<String> ls = new ArrayList<>();
        ArrayList<String> notepad = new ArrayList<>();
        ArrayList<String> timeHelp = new ArrayList<>();
        ArrayList<String> dateHelp = new ArrayList<>();
        
        about.add("about\t\tDisplay the author and the version number\n");
        cmdHelp.add("cmd\t\tOpen command prompt\n");
        cls.add("cls\t\tClear screen(only work in cmd)\n");
        exit.add("exit\t\tTerminate the program\n");
        help.add("help\t\tList down all the command and usage\n");
        hist.add("hist\t\t[-l | -c]");
        hist.add("-l x\t\tPrint out the previous x number of used commands");
        hist.add("-c\t\tClear the command history\n");
        ls.add("ls\t\t[no arg | -f | -d] [OPTIONAL file path]");
        ls.add("no arg\t\tPrint out both files and directory");
        ls.add("-f\t\tPrint out files");
        ls.add("-d\t\tPrint out directory\n");
        notepad.add("notepad\t\t[no arg | -t]");
        notepad.add("no arg\t\tOpen notepad");
        notepad.add("-t [title]\tCreate Notepad with the title entered\n");
        timeHelp.add("time\t\tDisplay time\n");
        dateHelp.add("date\t\tDisplay date\n");
        helpList.add(about);
        helpList.add(cmdHelp);
        helpList.add(cls);
        helpList.add(dateHelp);
        helpList.add(exit);
        helpList.add(help);
        helpList.add(hist);
        helpList.add(ls);
        helpList.add(notepad);
        helpList.add(timeHelp);
        
        int easteregg = 0;
        
        commandHistory.clear();
        
        COMMANDLIST.addAll(Arrays.asList(commandList));

        String commandLine;

        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("My Shell [Version 0.1]");
        System.out.println("(c) 2018 DianChia Corporation. All rights reserved.");

        while (true) {
            System.out.print("00000074> ");

            commandLine = console.readLine();
            ArrayList<String> commandParams = new ArrayList<>();
            String[] commandSplit = commandLine.split(" ");
            if (commandSplit.length > 1) {
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            }
            if (commandSplit[0].equals("")) {
                continue;
            }
            
            if(!COMMANDLIST.contains(commandSplit[0])){
                System.out.println("Invalid command!");
                System.out.println("Enter 'help' for list of commands.");
                continue;
            }
            
            commandHistory.add(commandLine);

            if (commandSplit[0].equalsIgnoreCase("time")) {
                if(commandParams.size() >= 1){
                    if(commandParams.get(0).equals("/?")){
                        for(String line : timeHelp){
                            System.out.println(line);
                        }
                    }
                }
                else{
                    Date time = new Date();
                    SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                    System.out.println(sdf.format(time));
                }
            }

            if (commandSplit[0].equalsIgnoreCase("date")) {
                if(commandParams.size() >= 1){
                    if(commandParams.get(0).equals("/?")){
                        for(String line : dateHelp){
                            System.out.println(line);
                        }
                    }
                }
                else{
                    Date date = new Date();
                    SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                    System.out.println(sdf.format(date));
                }
            }

            if(commandSplit[0].equalsIgnoreCase("about")){
                if(commandParams.size() >= 1){
                    if(commandParams.get(0).equals("/?")){
                        for(String line : about){
                            System.out.println(line);
                        }
                    }
                }
                else{
                    System.out.println("Author: Chia Dian Rui");
                    System.out.println("Version: 0.01");
                }
            }

            if (commandSplit[0].equalsIgnoreCase("ls")) {
                try{
                    File currentDir = new File("");
                    File fileList = null;
                    ArrayList <String> fileArray = new ArrayList<>();
                    if(commandParams.size() >= 1){
                        if(commandParams.get(0).matches("(-|[/]).*")){
                            if(commandParams.size() > 1){
                                for(int i = 1; i < commandParams.size(); i++){                        
                                    fileArray.add(commandParams.get(i));
                                }
                                String filePath = String.join(" ", fileArray);
                                fileList = new File(filePath);
                            }
                            else{
                                fileList = new File(currentDir.getAbsolutePath());
                            }
                            File[] files = fileList.listFiles();
                            switch (commandParams.get(0)) {
                                case "-f":
                                    for(File file : files){
                                        if(file.isFile()){
                                            System.out.println(file);
                                        }
                                    }   break;
                                case "-d":
                                    for(File file : files){
                                        if(file.isDirectory()){
                                            System.out.println(file);
                                        }
                                    }   break;
                                case "/?":
                                    for(String line : ls){
                                        System.out.println(line);
                                    }   break;
                                default:
                                    System.out.println("Invalid parameter!");
                                    System.out.println("Enter " + commandSplit[0] + " /? for more option");
                                    commandHistory.remove(commandLine);
                                    continue;
                            }
                        }
                        else{
                            if(commandParams.size() >= 1){
                                for(int i = 0; i < commandParams.size(); i++){                        
                                    fileArray.add(commandParams.get(i));
                                }
                                String filePath = String.join(" ", fileArray);
                                fileList = new File(filePath);
                            }
                            else{
                                fileList = new File(currentDir.getAbsolutePath());
                            }
                            File[] files = fileList.listFiles();
                            for(File file : files){
                                System.out.println(file);
                            }
                        }
                    }
                    else{
                        fileList = new File(currentDir.getAbsolutePath());
                        File[] files = fileList.listFiles();
                        for(File file : files){
                            System.out.println(file);
                        }
                    }        
                }
                catch(NullPointerException ex){
                    System.out.println("Path not found.");
                }
            }

            if (commandSplit[0].equalsIgnoreCase("hist")) {
                if (commandParams.size() < 1) {
                    System.out.println("Error : Not enough input argument(s). Please enter hist /? for help.");
                } else {
                    switch (commandParams.get(0)) {
                        case "-h":
                            hist.forEach((line)->
                            System.out.println(line));
                            break;
                        case "/?":
                            for(String line : hist){
                                System.out.println(line);
                            }   break;
                        case "-c":
                            commandHistory.clear();
                            System.out.println("Deleted: All command history.");
                            break;
                        case "-l":
                            String temp = commandParams.get(1);
                            int n = Integer.parseInt(temp);
                            int last = commandHistory.size();
                            int index = last - n;
                            if(index <= 1){
                                //System.out.println("There is only " + (commandHistory.size() - 1) + " command before.");
                                continue;
                            }   for(int i = index - 1; i < last - 1; i++){
                                System.out.println(commandHistory.get(i));
                            }   if(commandParams.size() > 2){
                                if(commandParams.get(2).equals("-c")){
                                    commandHistory.clear();
                                    System.out.println("Deleted: All command history.");
                                }
                            }   break;
                        default:
                            System.out.println("Invalid parameter!");
                            System.out.println("Enter " + commandSplit[0] + " /? for more option");
                            commandHistory.remove(commandLine);
                            continue;
                    }
                }
            }

            if (commandSplit[0].equalsIgnoreCase("notepad")){
                String title = "";
                if(commandParams.size() >= 1){
                    switch (commandParams.get(0)) {
                        case "-t":
                            title = commandParams.get(1);
                            ProcessBuilder pb = new ProcessBuilder("Notepad.exe", title);
                            pb.start();
                            break;
                        case "/?":
                            for(String line : notepad){
                                System.out.println(line);
                            }   break;
                        default:
                            System.out.println("Invalid parameter!");
                            System.out.println("Enter " + commandSplit[0] + " /? for more option");
                            commandHistory.remove(commandLine);
                            continue;
                    }
                }
                else{
                    ProcessBuilder pb = new ProcessBuilder("Notepad.exe", title);
                    pb.start();
                }
            }
            
            if(commandSplit[0].equalsIgnoreCase("cmd")){
                final List<String> cmd = new ArrayList<>();
                
                cmd.add("cmd.exe");
                cmd.add("/C");
                cmd.add("start");
                if(commandParams.size() >= 1){
                    if(commandParams.get(0).equals("/?")){
                        for(String line : cmdHelp){
                            System.out.println(line);
                        }
                    }
                }
                else{
                    ProcessBuilder pb = new ProcessBuilder(cmd);
                    pb.start();
                }
            }
            
            if(commandSplit[0].equalsIgnoreCase("help")){
                if(commandParams.size() >= 1){
                    if(commandParams.get(0).equals("/?")){
                        for(String line : help){
                            System.out.println(line);
                        }
                    }
                }
                else{
                    for(ArrayList<String> list : helpList){
                        ArrayList<String> lists = list;
                        for(String i : lists){
                            System.out.println(i);
                        }
                    }
                }
            }
            if(easteregg == 20){
                System.out.println("Congratulation!! You unlocked an easter egg XD.");
                System.out.println("Credit to Dian Rui(which is myself) for creating this.");
            }
            if(commandSplit[0].equalsIgnoreCase("cls")){
                if(commandParams.size() >= 1){
                    if(commandParams.get(0).equals("/?")){
                        for(String line : cls){
                            System.out.println(line);
                        }
                    }
                }
                else{
                    try{
                        new ProcessBuilder("cmd", "/c", "cls").inheritIO().start().waitFor();
                    }catch(InterruptedException ex){
                    }
                }
            }
            if (commandSplit[0].equalsIgnoreCase("exit")) {
                if(commandParams.size() >= 1){
                    if(commandParams.get(0).equals("/?")){
                        for(String line : exit){
                            System.out.println(line);
                        }
                    }
                }
                else
                    System.exit(0);
            }
            
            if(commandSplit[0].equalsIgnoreCase("haha")){
                easteregg++ ;
            }
        }
    }
}
