import java.io.*;
import java.util.ArrayList;
import java.util.Date;
import java.text.*;
import java.util.Iterator;
import java.util.Vector;

public class myshell_00000090 {
    public static void main(String[] args) throws IOException {

        //Use commandLine to store the command and the argument(s).
        String commandLine;

        //Read and buffered the command and the argument(s).
        BufferedReader console = new BufferedReader(new InputStreamReader(System.in));

        //Just a banner
        System.out.println("My Shell Version 0.01");
        
        Vector history = new Vector(); //vector to store previous history
        
        //Continue to loop until "exit" is entered.
        //Although it is uncommon for a normal application, it is normal for a shell.
        while (true) {
            //Print the console line.
            //Remember to replace "student_id" with your student number.
            System.out.print("00000090> ");

            //Read the input from user.
            commandLine = console.readLine();

            //Split the input into command and parameters.
            //commandParams is a list that is pre-created to store all the parameters.
            //commandSplit is a string array that is used to store the command and the parameters after spliting them.
            ArrayList commandParams = new ArrayList();
            String[] commandSplit = commandLine.split(" ");
            
            //Check to see if there is any parameter.
            //If size of commandSplit is larger than one that means the user has included some parameters in the input.
            if (commandSplit.length > 1) {
                //Add the parameters into commandParams
                for (int i = 1; i < commandSplit.length; i++) {
                    commandParams.add(commandSplit[i]);
                }
            } 

            //Reprint the conline line if user just pressed "enter" without any command.
            if (commandSplit[0].equals("")) {
                continue;
            } 
            
            //Print user name and student ID
            if (commandSplit[0].equals("about")){
                System.out.println("author: Chuah Ling Ming");
                System.out.println("version: 0.01");
                history.add("about");
            } 

            //Print the current time.
            else if (commandSplit[0].equals("time")) {
                Date time = new Date();
                //Format the output in hour:minute:second AM/PM format
                SimpleDateFormat sdf = new SimpleDateFormat("hh:mm:ss a");
                System.out.println(sdf.format(time));
                history.add("time");
            } 
            
            //Print today's date.
            else if (commandSplit[0].equals("date")) {
                Date date = new Date();
                //Format the output in day/month/year format
                SimpleDateFormat sdf = new SimpleDateFormat("E dd/MM/yyyy");
                System.out.println(sdf.format(date));
                history.add("date");
            } 
            
            //Exit the console when "exit" is entered.
            else if (commandSplit[0].equals("exit")) {
                System.exit(0);
            } 
                
            //To retrieve files and folders (or directories) from the current directory 
            else if (commandSplit[0].equals("ls") && commandSplit.length==1){
                //get all the files from a directory
                File directory = new File("C:\\Users\\Vymin Chuah\\Documents\\NetBeansProjects\\myshell\\src");
                if (commandSplit[0].equals("ls") && commandSplit.length==1){
                    File[] fList = directory.listFiles();
                    for (File file : fList){
                        System.out.println(file.getName()); //print all file and directory names
                    }
                } else
                    if (commandSplit[0].equals("ls") && commandParams.get(0).equals("-f")) {
                        String files;
                        File[] fList = directory.listFiles();
                        for (int i = 0; i < fList.length; i++) {
                            if (fList[i].isFile()) {
                                files = fList[i].getName();
                                System.out.println(files); //print all file names
                            }
                        } 
                    } else 
                        if (commandSplit[0].equals("ls") && commandParams.get(0).equals("-d")) {
                            String files;
                            File[] fList = directory.listFiles();
                            for (int i = 0; i < fList.length; i++) {
                                if (fList[i].isDirectory()) {
                                    files = fList[i].getName();
                                    System.out.println(files); //print all directory names
                                }
                            }
                        } 
            } 
                //Launch the notepad
                else  if (commandSplit[0].equals("notepad")) { 
                    ProcessBuilder pb = new ProcessBuilder("notepad.exe");
                    pb.start();
                } 
                else if (commandSplit[0].equals("hist")) { //start inserting input for history listing. must start with hist
                if (commandParams.size() < 1) { 
                    System.out.println("Error : Not enough input argument(s). Please enter hist -h for help.");
                } 
                    else if(commandParams.size()>=1 && commandParams.size()<=2){
                        //Process based on the number of parameters attached to the command.
                        if (commandParams.get(0).equals("-h")) { 
                            System.out.println("hist -l [int] : Retrieve the last [int] commands entered by the user.");
                            System.out.println("hist -c : Clear the command history.");
                        } else {
                            //when the input is anything other than integer, it prints Invalid number
                            if(commandParams.get(0).equals("-l") && Integer.parseInt(commandParams.get(1).toString())<= 0){
                                System.out.println("Invalid number");   
                            } else {
                                //list the history out according to user's input. 
                                //when hist -l 100 is input, all history will be printed.
                                if(commandParams.get(0).equals("-l") && history.size() < Integer.parseInt(commandParams.get(1).toString())){
                                    for (int count = 0; count < history.size(); count++){
                                        System.out.println(history.get(count));
                                    }
                                } else {
                                    //if there are 5 in the list and user only input 3,
                                    //then it will only list the last 3 history
                                    if(commandParams.get(0).equals("-l") && history.size() > Integer.parseInt(commandParams.get(1).toString())){
                                        for (int x = history.size()-1-Integer.parseInt(commandParams.get(1).toString()); x < history.size()-1; x++){
                                            System.out.println(history.get(x));
                                        }
                                    } else {
                                        //remove all previous history    
                                        if (commandParams.get(0).equals("-c")) {
                                            history.removeAll(history);
                                            System.out.println("Deleted: All command history.");
                                        } else 
                                            if (commandParams.get(0).equals("-c") && commandParams.get(1).equals("-l")){ 
                                                //it shows the result of delete after listing and vice versa
                                                //the result for this will return empty
                                                try{
                                                    int x;
                                                    x = Integer.parseInt(commandParams.get(2).toString());
                                                
                                                    if (x<1){
                                                        System.out.println("Invalid number");
                                                    } else {
                                                        System.out.println("All command history has been deleted.");
                                                        }
                                                }   catch (Exception e){
                                                    continue;
                                                    }
                                            }
                                        }
                                    }
                                }
                            }
                        }
            } else { 
                //if the wrong input is input, it prints Invalid command        
                System.out.println("Invalid command!");
                }
        } 
    }  
}
